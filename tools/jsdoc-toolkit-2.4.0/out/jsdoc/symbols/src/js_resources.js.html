1 /** 
  2  * @class
  3  * @constructor
  4  * Resources loader
  5  */
  6 function Resources() {
  7 }
  8 
  9 /** 
 10  * @field
 11  * @static
 12  * Resources configuration must be configured per application to be able 
 13  * to locate resources.
 14  */
 15 Resources.configuration = {
 16 	ROOT_PATH : "../",
 17 	IMG_PATH : "../img/"
 18 };
 19 
 20 /**
 21  * @method
 22  * @static
 23  * Loads an image resource from name
 24  * TODO: An idea is to later combine all images into a sprite sheet and this method
 25  *       could possibly retrieve it from the spritesheet rather than file.
 26  * @return Image The loaded image
 27  */
 28 Resources.loadImage = function(name) {
 29 	var image = new Image();
 30 	image.src = Resources.configuration.IMG_PATH + name;
 31 	return image;
 32 }
