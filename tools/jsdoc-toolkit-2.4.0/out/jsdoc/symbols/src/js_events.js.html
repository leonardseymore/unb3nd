  2 /**
  3  * @class
  4  * @constructor
  5  * @abstract
  6  * An observable object that can raise events to interrested listeners
  7  */
  8 function Observable() {
  9 
 10 	/**
 11 	 * @field EventListener []
 12 	 * All registered event listeners
 13 	 */
 14 	this.listeners = new Object();
 15 	
 16 	/**
 17 	 * @method
 18 	 * Add an event listener of the specified type
 19 	 * @param string type The type of event to listen for
 20 	 * @param function listener The listener to invoke
 21 	 */
 22 	this.addEventListener = function(type, listener) {
 23 		if (typeof this.listeners[type] == "undefined") {
 24 			this.listeners[type] = [];
 25 		} // if
 26 		this.listeners[type].push(listener);
 27 	}
 28 	
 29 	/**
 30 	 * @method
 31 	 * Dispatch an event of specified type to all interrested listeners
 32 	 * @param string type The type of event to listen for
 33 	 * @param e The actual event
 34 	 */
 35 	this.dispatchEvent = function(type, e) {
 36 		for (i in this.listeners[type]) {
 37 			var listener = this.listeners[type][i];
 38 			listener.call(this, e);
 39 		} // for
 40 	}
 41 	
 42 	/**
 43 	 * @method
 44 	 * Removes the specified event listener
 45 	 * @param string type The type of event to listen for
 46 	 * @param function listener The listener to remove
 47 	 */
 48 	this.removeEventListener = function(type, listener) {
 49 		if (this.listeners[type] instanceof Array){
 50             var listeners = this.listeners[type];
 51             for (var i=0, len=listeners.length; i < len; i++){
 52                 if (listeners[i] === listener){
 53                     listeners.splice(i, 1);
 54                     break;
 55                 } // if
 56             } // for
 57         } // if
 58 	}
 59 }
