1 /**
  2  * @class
  3  * @constructor
  4  * A queue (FIFO) collection implementation
  5  */
  6 function Queue() {
  7 
  8 	/** 
  9 	 * @field Object []
 10 	 * Array place holder for queue items
 11 	 */
 12 	var queue = [];
 13 	
 14 	/**
 15 	 * @method
 16 	 * Gets the size of the queue
 17 	 * @return int The size of the queue
 18 	 */
 19 	this.getSize = function() {
 20 		return queue.length;
 21 	}
 22 	
 23 	/**
 24 	 * @method
 25 	 * Determines if the queue is empty
 26 	 * @return boolean True if the queue is empty
 27 	 */
 28 	this.isEmpty = function() {
 29 		return (queue.length == 0);
 30 	}
 31 	
 32 	/**
 33 	 * @method
 34 	 * Enqueues the specified element in this queue
 35 	 * @param element The element to push into the queue
 36 	 * @return int The new length of the queue
 37 	 */
 38 	this.enqueue = function(element) {
 39 		return queue.push(element);
 40 	}
 41 	
 42 	/**
 43 	 * @method
 44 	 * Dequeues the oldest element in the queue
 45 	 * @return Object The oldest element in the queue, null if queue is empty
 46 	 */
 47 	this.dequeue = function() {
 48 		return queue.shift();
 49 	}
 50 	
 51 	/**
 52 	 * @method
 53 	 * Gets all elements in this queue
 54 	 * @return Object [] All elements in the queue
 55 	 */
 56 	this.getElements = function() {
 57 		return queue;
 58 	}
 59 }

 61 /**
 62  * @class
 63  * @constructor
 64  * A ringbuffer(FIFO) collection implementation.  The ringbuffer
 65  * will automatically push out old items when maximum size is reached.
 66  * @param int size The size of the buffer
 67  */
 68 function RingBuffer(size) {
 69 	
 70 	/**
 71 	 * Size must be passed in
 72 	 */
 73 	if (typeof(size) == "undefined") {
 74 		throw new Error("Valid integer size expected");
 75 	} // if
 76 
 77 	/** 
 78 	 * @field Queue
 79 	 * Array place holder for queue items
 80 	 */
 81 	var queue = new Queue();
 82 	
 83 	/**
 84 	 * @method
 85 	 * Gets the size of the queue
 86 	 * @return int The size of the queue
 87 	 */
 88 	this.getSize = function() {
 89 		return queue.getSize();
 90 	}
 91 	
 92 	/**
 93 	 * @method
 94 	 * Determines if the queue is empty
 95 	 * @return boolean True if the queue is empty
 96 	 */
 97 	this.isEmpty = function() {
 98 		return queue.isEmpty();
 99 	}
100 	
101 	/**
102 	 * @method
103 	 * Enqueues the specified element in this queue
104 	 * @param element The element to push into the queue
105 	 * @return int The new length of the queue
106 	 */
107 	this.enqueue = function(element) {
108 		if (this.getSize() >= size) {
109 			queue.dequeue();
110 		} // if
111 		
112 		return queue.enqueue(element);
113 	}
114 	
115 	/**
116 	 * @method
117 	 * Dequeues the oldest element in the queue
118 	 * @return Object The oldest element in the queue, null if queue is empty
119 	 */
120 	this.dequeue = function() {
121 		return queue.dequeue();
122 	}
123 	
124 	/**
125 	 * @method
126 	 * Gets all elements in this buffer
127 	 * @return Object [] All elements in the buffer
128 	 */
129 	this.getElements = function() {
130 		return queue.getElements();
131 	}
132 }

135  * @class
136  * @constructor
137  * A set of unique elements
138  */
139 function Set() {
140 	/**
141 	 * @field
142 	 * All elements in the set
143 	 */
144 	this.elements = [];
145 	
146 	/**
147 	 * @method
148 	 * Add a new element to the set
149 	 * @param Object element The element to add
150 	 * @return int Number of elements
151 	 */
152 	this.add = function(element) {
153 		return this.elements.push(element);
154 	}
155 	
156 	/**
157 	 * @method
158 	 * Remove an element at the specified index
159 	 * @param index The index of the element
160 	 * @return Object The removed element
161 	 */
162 	this.removeEntityAtIndex = function(index) {
163 		return this.elements.splice(index, 1);
164 	}
165 	
166 	/**
167 	 * @method
168 	 * Remove the specified element from the set
169 	 * @param Object element The element to remove
170 	 * @return boolean True if the element was removed
171 	 */
172 	this.remove = function(element) {
173 		for (i in this.elements) {
174 			var el = this.elements[i];
175 			if (el === element) {
176 				this.removeElementAtIndex(i);
177 				return true;
178 			} // if
179 		} // for
180 		return false;
181 	}
