  5 /**
  6  * @class
  7  * @constructor
  8  * @abstract
  9  * Background base class
 10  */
 11 function Background() {
 12 
 13 	/**
 14 	 * @method
 15 	 * @abstract
 16 	 * Update the background
 17 	 * @arg int delta Delta time in milliseconds
 18 	 * @return void
 19 	 */
 20 	this.update = function(delta) {}
 21 	
 22 	/**
 23 	 * @method
 24 	 * @abstract
 25 	 * Draw the background
 26 	 * @arg Rectangle area The area to draw the background on
 27 	 * @return void
 28 	 */
 29 	this.draw = function(area) {}
 30 }

 32 /**
 33  * @class
 34  * @constructor
 35  * A simple background that takes a fill style for the background
 36  * @param FillStyle The style to use to fill the background
 37  */
 38 function FillBackground(fillStyle) {
 39 
 40 	/**
 41 	 * @field FillStyle
 42 	 * Fill style to flood the background
 43 	 */
 44 	this.fillStyle = fillStyle;
 45 	
 46 	/**
 47 	 * @method
 48 	 * @override
 49 	 * Draw the background
 50 	 * @arg Rectangle area The area to draw the background on
 51 	 * @return void
 52 	 */
 53 	this.draw = function(area) {
 54 		ctx.fillStyle = this.fillStyle;
 55 		ctx.fillRect(area.x, area.y, area.width, area.height);
 56 	}
 57 }
 58 FillBackground.prototype = new Background();

 60 /**
 61  * @class
 62  * @constructor
 63  * A starry sky background
 64  * @param int numStars Number of stars
 65  * @param FillStyle fillStyle Basic fill style
 66  */
 67 function StarryBackground(numStars, fillStyle) {
 68 
 69 	/**
 70 	 * @class
 71 	 * @constructor
 72 	 * A star
 73 	 * @param int x X-coordinate
 74 	 * @param int y Y-coordinate
 75 	 * @param float i Intensity 0 to 1
 76 	 */
 77 	function Star(x, y, i) {
 78 		/**
 79 		 * @field int
 80 		 * X-coordinate
 81 		 */
 82 		this.x = x;
 83 		
 84 		/**
 85 		 * @field int
 86 		 * Y-coordinate
 87 		 */
 88 		this.y = y;
 89 		
 90 		/**
 91 		 * @field float
 92 		 * Intensity 0 to 1
 93 		 */
 94 		this.i = i;
 95 		
 96 		/**
 97 		 * @field float
 98 		 * Radius 
 99 		 */
100 		this.r = Math.floor(Math.random() * 4) + 2;
101 		
102 		/**
103 		 * @method
104 		 * Animate this star
105 		 * @param int delta Delta time in milliseconds
106 		 */
107 		this.animate = function(delta) {
108 			if (Math.random() < 0.005) {
109 				this.i = Math.random();
110 			} // if
111 		}
112 		
113 		/**
114 		 * @method
115 		 * Draw the star as a pixel
116 		 * @return void
117 		 */
118 		this.draw = function() {
119 			ctx.fillStyle = "rgba(255, 255, 255, " + this.i + ")";
120 			ctx.fillRect(this.x, this.y, 1, 1);
121 		}
122 		
123 		/**
124 		 * @method
125 		 * Draw the star as a cartoon styled star
126 		 * @return void
127 		 */
128 		this.drawCartoon = function() {
129 			ctx.save();
130 			ctx.fillStyle = "rgba(255, 255, 255, " + this.i + ")";
131 			ctx.translate(this.x, this.y);
132 			ctx.beginPath()
133 			ctx.moveTo(this.r, 0);
134 			for (var i = 0; i < 9; i++){
135 				ctx.rotate(Math.PI / 5);
136 				if(i % 2 == 0) {
137 					ctx.lineTo((this.r / 0.525731) * 0.200811,0);
138 				} else {
139 					ctx.lineTo(this.r, 0);
140 				} // if
141 			} // for
142 			ctx.closePath();
143 			ctx.fill();
144 			ctx.restore();
145 		}
146 	}
147 	
148 	/**
149 	 * @field boolean
150 	 * Cartoonify stars
151 	 */
152 	this.cartoonify = true;
153 	
154 	/**
155 	 * @field FillStyle
156 	 * Fill style to flood the background
157 	 */
158 	this.fillStyle = (fillStyle == undefined ? STYLE_DARK_EVENING : fillStyle);
159 	
160 	/**
161 	 * @field int
162 	 * Number of stars
163 	 */
164 	this.numStars = (numStars == undefined ? 50 : numStars);
165 	
166 	/**
167 	 * @field Star
168 	 * Very lucky if you see one of these
169 	 */
170 	this.shootingStar = null;
171 	
172 	/**
173 	 * @field float []
174 	 * Star intensity
175 	 */
176 	this.stars = new Array();
177 	for (i = 0; i < numStars; i++) {
178 		var star = new Star(windowRect.width * Math.random(), 
179 			windowRect.height * Math.random(), 
180 			Math.random());
181 		this.stars.push(star);
182 	}
183 	
184 	/**
185 	 * @method
186 	 * @override
187 	 * Update the background
188 	 * @arg int delta Delta time in milliseconds
189 	 * @return void
190 	 */
191 	this.update = function(delta) {
192 		var dt = delta / 1000;
193 		if (!this.shootingStar) {
194 			if (Math.random() < 0.002) {
195 				this.shootingStar = new Star(0, Math.random() * windowRect.height, Math.random());
196 			} // if
197 		} else {
198 			this.shootingStar.x += 500 * dt;
199 			this.shootingStar.y += 300 * dt;
200 			if (this.shootingStar.x > windowRect.width) {
201 				this.shootingStar = undefined;
202 			} // if
203 		} // if
204 		
205 		for (i = 0; i < numStars; i++) {
206 			var star = this.stars[i];
207 			star.animate(delta);
208 		} // for
209 	}
210 	
211 	/**
212 	 * @method
213 	 * @override
214 	 * Draw the background
215 	 * @arg Rectangle area The area to draw the background on
216 	 * @return void
217 	 */
218 	this.draw = function(area) {
219 		ctx.save();
220 		ctx.fillStyle = this.fillStyle;
221 		ctx.fillRect(area.x, area.y, area.width, area.height);
222 		for (i = 0; i < numStars; i++) {
223 			var star = this.stars[i];
224 			if (this.cartoonify) {
225 				star.drawCartoon();
226 			} else {
227 				star.draw();
228 			} // if
229 		} // for
230 		
231 		if (this.shootingStar) {
232 			if (this.cartoonify) {
233 				this.shootingStar.drawCartoon();
234 			} else {
235 				this.shootingStar.draw();
236 			} // if
237 		} // if		
238 		ctx.restore();
239 	}
240 }
241 StarryBackground.prototype = new Background();
