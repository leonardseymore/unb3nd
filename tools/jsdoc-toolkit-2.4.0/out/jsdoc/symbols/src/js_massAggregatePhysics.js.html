  5 /**
  6  * @class
  7  * @constructor
  8  * Particle is an object which has a position but no orientation
  9  */
 10 function Particle() {
 11 
 12 	/**
 13 	 * @super
 14 	 */
 15 	Observable.call(this);
 16 	
 17 	/**
 18 	 * @field long
 19 	 * Unique identifier to help track a single particle
 20 	 */
 21 	this.uid = Particle.getNextUid();
 22 
 23 	/**
 24 	 * @field Vector2
 25 	 * The position
 26 	 */
 27 	this.pos = new Vector2();
 28 	
 29 	/**
 30 	 * @field Vector2
 31 	 * The velocity
 32 	 */
 33 	this.vel = new Vector2();
 34 	
 35 	/**
 36 	 * @field Vector2
 37 	 * The acceleration
 38 	 */
 39 	this.acc = new Vector2();
 40 	
 41 	/**
 42 	 * @field float
 43 	 * Damping is a simple yet special property involved in slowing down moving objects
 44 	 * 0 - 1, where 0 is full damping and 1 is no damping
 45 	 */
 46 	this.damping = 1;
 47 	
 48 	/**
 49 	 * @field float
 50 	 * Holds the inverse mass of the particle.
 51 	 */
 52 	this.inverseMass = 0;
 53 	
 54 	/**
 55 	 * @field Vector2
 56 	 * The overall force accumulator
 57 	 */
 58 	this.forceAccum = new Vector2();
 59 	
 60 	/**
 61 	 * @method
 62 	 * Determines if point is near this particle
 63 	 * @param Vector2 point Point to test
 64 	 * @param float distance Distance to test
 65 	 * @return boolean true If supplied point is near this particle
 66 	 */
 67 	 this.isCloseToPoint = function(point, distance) {
 68 		return this.pos.sub(point).getMagnitude() <= distance;
 69 	 }
 70 	
 71 	/**
 72 	 * @method
 73 	 * Ensures this body has a finite mass
 74 	 * @return boolean true If this body has a finite mass
 75 	 */
 76 	 this.hasFiniteMass = function() {
 77 		return this.inverseMass > 0;
 78 	 }
 79 	
 80 	/**
 81 	 * @method
 82 	 * An external force to be added on the particle
 83 	 * @param Vector2 o The force to add to this particle
 84 	 * @return void
 85 	 */
 86 	this.applyForce = function(force) {
 87 		this.forceAccum.addMutate(force);
 88 	}
 89 	
 90 	/**
 91 	 * @method
 92 	 * Reset the overall force on this particle
 93 	 * @return void
 94 	 */
 95 	this.clearForceAccum = function() {
 96 		this.forceAccum.zeroMutate();
 97 	}
 98 	
 99 	/**
100 	 * @method
101 	 * Sets the inverse mass of the particle
102 	 * @param float inverseMass The inverse mass of the particle
103 	 * @return void
104 	 */
105 	this.setInverseMass = function(inverseMass) {
106 		this.inverseMass = inverseMass;
107 	}
108 	
109 	/**
110 	 * @method
111 	 * Sets the mass of the particle
112 	 * @param float mass The mass of the particle
113 	 * @return void
114 	 */
115 	this.setMass = function(mass) {
116 		if (mass == 0) {
117 			this.inverseMass = 0;
118 		} else {
119 			this.inverseMass = 1 / mass;
120 		} // if
121 	}
122 	
123 	/**
124 	 * @method
125 	 * Gets the mass of the particle
126 	 * @return float mass The mass of the particle
127 	 */
128 	this.getMass = function() {
129 		if (this.inverseMass == 0) {
130 			return 0;
131 		} else {
132 			return 1 / this.inverseMass;
133 		} // if
134 	}
135 	
136 	/**
137 	 * @method
138 	 * Integrates the particle forward in time by the given amount.
139 	 * Uses Newton-Euler integration method
140 	 * @param int delta The time delta in milliseconds
141 	 * @return void
142 	 */
143 	this.integrate = function(delta) {		
144 		var dt = delta / 1000;
145 		
146 		this.pos.addMutate(
147 			this.vel.multScalar(dt)
148 		);
149 
150 		var resultingAcc = this.acc.add(
151 			this.forceAccum.multScalar(this.inverseMass)
152 		);
153 		
154 		this.vel.addMutate(
155 			resultingAcc.multScalar(dt)
156 		);
157 		
158 		this.vel.multScalarMutate(Math.pow(this.damping, dt));
159 		
160 		this.clearForceAccum();
161 	}
162 	
163 	/**
164 	 * @method
165 	 * Draw visual helpers for this vector
166 	 *
167 	 * This method assumes that translations have already been made to the starting
168 	 * position of the vector
169 	 * @param fillStyle Optional overriding fill style
170 	 * @return void
171 	 */
172 	this.draw = function(fillStyle) {
173 		ctx.save();	
174 		ctx.fillStyle = (fillStyle == undefined ? SETTINGS.PARTICLE_COLOR : fillStyle);
175 		ctx.translate(this.pos.x, this.pos.y);
176 		var width = SETTINGS.PARTICLE_WIDTH;
177 		ctx.fillRect(
178 			-width, -width, width * 2, width * 2
179 		);
180 		ctx.restore();
181 	}
182 	
183 	/**
184 	 * @method
185 	 * Draw string representation of particle
186 	 * @param fillStyle Optional overriding fill style
187 	 * @return void
188 	 */
189 	this.drawToString = function(fillStyle) {
190 		ctx.save();	
191 		ctx.fillStyle = (fillStyle == undefined ? SETTINGS.PARTICLE_STRING_COLOR : fillStyle);
192 		ctx.translate(this.pos.x, this.pos.y);
193 		var width = SETTINGS.PARTICLE_WIDTH;
194 		ctx.fillText(
195 			this.toString(),width, width
196 		);
197 		ctx.restore();
198 	}
199 	
200 	/**
201 	 * @field
202 	 * On die callback
203 	 */
204 	this.ondie = undefined;
205 
206 	/**
207 	 * @method
208 	 * Die callback invoker
209 	 * @return void
210 	 */
211 	this.die = function() {		
212 		if (this.ondie) {
213 			this.ondie();
214 		} // if
215 		this.dispatchEvent("die");
216 	}
217 	
218 	/**
219 	 * @method
220 	 * @abstract
221 	 * Converts the class to a string representation
222 	 * @return string The string representation of this class
223 	 */
224 	this.toString = function() {
225 		return "uid=" + this.uid;
226 	}
227 }

249  * @class
250  * @constructor
251  * A force registry for matching up force generators to particles
252  */
253 function ParticleForceRegistry() {
254 	
255 	/**
256 	 * @field {"particle":Particle, "forceGenerators":ForceGenerator []};
257 	 * Registry entries
258 	 */
259 	this.entries = [];
260 	
261 	/**
262 	 * @method
263 	 * Adds the force generator to the supplied particle
264 	 * TODO: This is a very brute force method, we need a HashMap or similar datastructure
265 	 * @param Particle particle The particle to add a force generator to
266 	 * @param ForceGenerator forceGenerator The force generator to add to the particle
267 	 * @return void
268 	 */
269 	this.add = function(particle, forceGenerator) {
270 		var entry;
271 		var found = false;
272 		for (i in this.entries) {
273 			entry = this.entries[i];
274 			if (entry.particle === particle) {
275 				found = true;
276 				break;
277 			} // if
278 		} // for
279 		
280 		if (!found) {
281 			entry = {"particle":particle, "forceGenerators":new Array(forceGenerator)};
282 			this.entries.push(entry);
283 		} else {
284 			entry.forceGenerators.push(forceGenerator);
285 		} // if
286 	}
287 	
288 	/**
289 	 * @method
290 	 * Gets all force generators for the supplied particle
291 	 * @return ForceGenerator [], undefined if not found
292 	 */
293 	this.getForceGenerators = function(particle) {
294 		for (i in this.entries) {
295 			var entry = this.entries[i];
296 			if (entry.particle === particle) {
297 				return entry.forceGenerators;
298 			} // if
299 		} // for
300 		
301 		return undefined;
302 	}
303 	
304 	/**
305 	 * @method
306 	 * Gets all force generators for the supplied particle
307 	 * @return ForceGenerator [], undefined if not found
308 	 */
309 	this.removeForceGenerators = function(particle) {
310 		for (i in this.entries) {
311 			var entry = this.entries[i];
312 			if (entry.particle === particle) {
313 				this.entries.splice(i, 1);
314 			} // if
315 		} // for
316 	}
317 	
318 	/**
319 	 * @method
320 	 * Removes a specific force generator
321 	 * @param ForceGenerator forceGenerator The specific force generator to remove
322 	 * @return ForceGenerator The removed force generator, undefined if nothing was removed
323 	 */
324 	this.removeForceGenerator = function(forceGenerator) {
325 		for (i in this.entries) {
326 			var entry = this.entries[i];
327 			for (j in entry.forceGenerators) {
328 				var fg = entry.forceGenerators[j];
329 				if (fg === forceGenerator) {
330 					return entry.forceGenerators.splice(j, 1);
331 				} // if
332 			} // if
333 		} // for
334 		return undefined;
335 	}
336 	
337 	/**
338 	 * @method
339 	 * Applies all forces for each particle over the time delta
340 	 * @param int delta Delta time in milliseconds
341 	 * @return void
342 	 */
343 	this.applyForces = function(delta) {
344 		for (i in this.entries) {
345 			var entry = this.entries[i];
346 			var particle = entry.particle;
347 			var forceGenerators = entry.forceGenerators;
348 			for (j in forceGenerators) {
349 				var forceGenerator = forceGenerators[j];
350 				forceGenerator.applyForce(particle, delta);
351 			} // for
352 		} // for
353 	}
354 	
355 	/**
356 	 * @method
357 	 * Draws visual helpers of all force generators
358 	 * @return void
359 	 */
360 	this.drawForceGenerators = function() {
361 		for (i in this.entries) {
362 			var entry = this.entries[i];
363 			var particle = entry.particle;
364 			var forceGenerators = entry.forceGenerators;
365 			for (j in forceGenerators) {
366 				var forceGenerator = forceGenerators[j];
367 				forceGenerator.draw(particle);
368 			} // for
369 		} // for
370 	}
371 }

417 function ParticleGravityForceGenerator() {
418 
419 	/**
420 	 * @field Vector2
421 	 * The gravitational pull
422 	 */
423 	this.gravitation = new Vector2(0, -9.81);
424 	
425 	/**
426 	 * @method
427 	 * Apply gravity to the given mass over the delta time
428 	 * @param Mass mass The mass to apply the force to
429 	 * @param int delta The delta time in milliseconds
430 	 */
431 	this.applyForce = function(particle, delta) {
432 		if (!particle.hasFiniteMass()) {
433 			if (debug && verbose) {
434 				console.debug("Particle %o has zero mass", particle);
435 			} // if
436 			return;
437 		} // if
438 		
439 		particle.applyForce(
440 			// scale force, since gravity is constant regardless of mass 
441 			this.gravitation.multScalar(particle.getMass())
442 		);
443 	}
444 	
445 	/**

455 		ctx.beginPath();
456 		ctx.translate(particle.pos.x, particle.pos.y);
457 		ctx.moveTo(0, 0);
458 		var gravityVector = this.gravitation.multScalar(particle.getMass())
459 		ctx.lineTo(gravityVector.x, gravityVector.y);
460 		ctx.stroke();

462 	}

475 /**
476  * @class 
477  * @constructor
478  * @extend ParticleForceGenerator
479  * A drag force generator
480  * @param float k1 Velocity drag coefficient
481  * @param float k2 Velocity squared drag coefficient
482  */
483 function ParticleDragForceGenerator(k1, k2) {
484 
485 	/**
486 	 * @field float
487 	 * Velocity drag coefficient
488 	 */
489 	this.k1 = (k1 == undefined ? 0 : k1);
490 	
491 	/**
492 	 * @field float
493 	 * Velocity squared drag coefficient
494 	 */
495 	this.k2 = (k2 == undefined ? 0 : k2);
496 	
497 	/**
498 	 * @method
499 	 * Apply gravity to the given mass over the delta time
500 	 * @param Mass mass The mass to apply the force to
501 	 * @param int delta The delta time in milliseconds
502 	 */
503 	this.applyForce = function(particle, delta) {
504 		var force = this.calculateForce(particle, delta);
505 		particle.applyForce(force);
506 	}
507 	
508 	/**
509 	 * @method
510 	 * Calculate the force for the supplied particle over the time delta
511 	 * @param Particle particle The particle to apply the force to
512 	 * @param int delta The delta time in milliseconds
513 	 * @return Vector2 The force to apply to this particle
514 	 */
515 	this.calculateForce = function(particle, delta) {
516 		var dragCoeff = particle.vel.getMagnitude();
517 		dragCoeff = this.k1 * dragCoeff + this.k2 * dragCoeff * dragCoeff;
518 		
519 		var force = particle.vel.normalize();
520 		force.multScalarMutate(-dragCoeff);
521 		return force;
522 	}
523 	
524 	/**
525 	 * @method
526 	 * @override
527 	 * Draw a visual representation of this force generator
528 	 * @param Particle particle The particle the generator is working on
529 	 * @return void
530 	 */
531 	this.draw = function(particle) {
532 		ctx.save();
533 		ctx.strokeStyle = SETTINGS.DRAG_COLOR;
534 		ctx.beginPath();
535 		ctx.translate(particle.pos.x, particle.pos.y);
536 		ctx.moveTo(0, 0);
537 		var dragVector = this.calculateForce(particle);
538 		ctx.lineTo(dragVector.x, dragVector.y);
539 		ctx.stroke();
540 		ctx.restore();
541 	}
542 	
543 	/**
544 	 * @method
545 	 * Converts the class to a string representation
546 	 * @return string The string representation of this class
547 	 */
548 	this.toString = function() {
549 		return "Drag: k1=" + this.k1 + " ,k2=" + this.k2;
550 	}

1050 /**
1051  * @class
1052  * @constructor
1053  * A contact represents two objects in contact (in this case
1054  * ParticleContact representing two particles). Resolving a
1055  * contact removes their interpenetration, and applies sufficient
1056  * impulse to keep them apart. Colliding bodies may also rebound.
1057  *
1058  * The contact has no callable functions, it just holds the
1059  * contact details. To resolve a set of contacts, use the particle
1060  * contact resolver class.
1061  */
1062 function ParticleContact() {
1063 
1064 	/**
1065 	 * @field Particle []
1066 	 * Holds the particles that are involved in the contact. The second may be
1067 	 * omitted in the case of a contact with the environment.
1068 	 */
1069 	this.particles = [];
1070 	
1071 	/**
1072 	 * @field float
1073 	 * Holds the normal restitution coefficient at the contact
1074 	 */
1075 	this.restitution = 0;
1076 	
1077 	/**
1078 	 * @field Vector2
1079 	 * Holds the contact normal
1080 	 */
1081 	this.contactNormal = new Vector2();
1082 	
1083 	/**
1084 	 * @field float
1085 	 * Penetration depth in the direction of the contact normal
1086 	 */
1087 	this.penetration = 0;
1088 	
1089 	/**
1090 	 * @method
1091 	 * @protected
1092 	 * Resolves this contact for both veolcity and interpenetration
1093 	 * @param int delta Delta time in milliseconds since last update
1094 	 * @return void
1095 	 */
1096 	this.resolve = function(delta) {
1097 		this.resolveVelocity(delta);
1098 		this.resolveInterpenetration(delta);
1099 	}
1100 	
1101 	/**
1102 	 * @method
1103 	 * @protected
1104 	 * Calculate separating velocity at this contact
1105 	 * @return float Separating velocity at this contact
1106 	 */
1107 	this.calculateSeparatingVelocity = function() {
1108 		var relativeVelocity = this.particles[0].vel.clone();
1109 		if (this.particles[1]) {
1110 			relativeVelocity.subMutate(
1111 				this.particles[1].vel
1112 			);
1113 		} // if
1114 		return relativeVelocity.dotProduct(this.contactNormal);
1115 	}
1116 	
1117 	/**
1118 	 * @method
1119 	 * @private
1120 	 * Handles impulse calculations for this collision
1121 	 * @param int delta Delta time in milliseconds since last update
1122 	 * @return void
1123 	 */
1124 	this.resolveVelocity = function(delta) {
1125 		var dt = delta / 1000;
1126 		var separatingVelocity = this.calculateSeparatingVelocity();
1127 		if (separatingVelocity > 0) {
1128 			// contact is either separating or stationary
1129 			return;
1130 		} // if
1131 		var newSepVelocity = -separatingVelocity * this.restitution;
1132 		
1133 		// ~PHYLIMIT keep to objects in resting contact
1134 		// micro-collision approach
1135 		// check velocity build up due to acceleration only
1136 		var accCausedVelocity = this.particles[0].acc.clone();
1137 		if (this.particles[1]) {
1138 			accCausedVelocity.subMutate(this.particles[1].acc);
1139 		} // if
1140 		var accCausedSepVelocity = accCausedVelocity.dotProduct(this.contactNormal) * dt;
1141 		
1142 		// If weve got a closing velocity due to acceleration build-up,
1143 		// remove it from the new separating velocity.
1144 		if (accCausedSepVelocity < 0) {
1145 			newSepVelocity += resitution * accCausedSepVelocity;
1146 			
1147 			// Make sure we havent removed more than was there to remove.
1148 			if (newSepVelocity < 0) {
1149 				newSepVelocity = 0;
1150 			} // if
1151 		} // if
1152 		// ~END-PHYLIMIT
1153 		
1154 		var deltaVelocity = newSepVelocity - separatingVelocity;
1155 		
1156 		var totalInverseMass = this.particles[0].inverseMass;
1157 		if (this.particles[1]) {
1158 			totalInverseMass += this.particles[1].inverseMass;
1159 		} // if
1160 		
1161 		if (totalInverseMass <= 0) {
1162 			// all particles have infinite mass
1163 			return;
1164 		} // if
1165 		
1166 		var impulse = deltaVelocity / totalInverseMass;
1167 		var impulsePerIMass = this.contactNormal.multScalar(impulse);
1168 		
1169 		// apply impulses proportional to inverse mass
1170 		// in the direction of the contact
1171 		this.particles[0].vel.addMutate(
1172 			impulsePerIMass.multScalar(this.particles[0].inverseMass)
1173 		);
1174 		
1175 		if (this.particles[1]) {
1176 			this.particles[1].vel.addMutate(
1177 				impulsePerIMass.multScalar(-this.particles[1].inverseMass)
1178 			);
1179 		} // if
1180 	}
1181 	
1182 	/**
1183 	 * @method
1184 	 * Handles interpenetration resolution for this contact
1185 	 * @param int delta Delta time in milliseconds since last update
1186 	 * @return void
1187 	 */
1188 	this.resolveInterpenetration = function(delta) {
1189 		if (this.penetration <= 0) {
1190 			return;
1191 		} // if
1192 		
1193 		var totalInverseMass = this.particles[0].inverseMass;
1194 		if (this.particles[1]) {
1195 			totalInverseMass += this.particles[1].inverseMass;
1196 		} // if
1197 		
1198 		if (totalInverseMass <= 0) {
1199 			// infinite mass
1200 			return;
1201 		} // if
1202 		
1203 		// find penetration resolution relative to inverse mass
1204 		var movePerIMass = this.contactNormal.multScalar(
1205 			-this.penetration / totalInverseMass
1206 		);
1207 		
1208 		this.particles[0].pos.addMutate(
1209 			movePerIMass.multScalar(this.particles[0].inverseMass)
1210 		);
1211 		
1212 		if (this.particles[1]) {
1213 			this.particles[1].pos.addMutate(
1214 				movePerIMass.multScalar(this.particles[1].inverseMass)
1215 			);
1216 		} // if
1217 	}
1218 }

1220 /**
1221  * @class
1222  * @constructor
1223  * The contact resolution routine for particle contacts. One resolver
1224  * instance can be shared for the whole simulation.
1225  */
1226 function ParticleContactResolver() {
1227 	/**
1228 	 * @field uint
1229 	 * Number of allowable iterations
1230 	 * Recommend: Minimally number of contacts, as a single contact is resolved
1231 	 *            per iteration.
1232 	 */
1233 	this.maxIt = 0;
1234 
1235 	/**
1236 	 * @method
1237 	 * Resolves the set of particle contacts for both penetration and velocity
1238 	 * @param ParticleContact [] contacts All contacts to resolve
1239 	 * @param int delta Delta time in milliseconds since last update
1240 	 * @return void
1241 	 */
1242 	this.resolveContacts = function(contacts, delta) {
1243 		var it = 0;
1244 		while (it < this.maxIt) {
1245 			// find the contact with the largest closing velocity
1246 			var max = 0;
1247 			var maxContact = contacts[0];
1248 			for (i in contacts) {
1249 				var contact = contacts[i];
1250 				var sepVel = contact.calculateSeparatingVelocity();
1251 				if (sepVel < max) {
1252 					max = sepVel;
1253 					maxContact = contact;
1254 				} // if
1255 			} // for
1256 			maxContact.resolve(delta);
1257 			it++;
1258 		} // while
1259 	}
1260 }

1530 		var usedContacts = 0;

1532 			if (usedContacts >= limit) {
1533 				break;
1534 			} // if
1535 			
1536 			var particle = this.particles[i];
1537 			for (j in this.particles) {
1538 				if (i == j) {
1539 					continue;
1540 				} // if
1541 				
1542 				var particleOther = this.particles[j];
1543 				var normal = particle.pos.sub(particleOther.pos);
1544 				var distance = normal.getMagnitude();
1545 				if (distance < this.collisionRadius) {
1546 					normal.normalizeMutate();
1547 					var particleContact = new ParticleContact();
1548 					particleContact.particles[0] = particle;
1549 					particleContact.particles[1] = particleOther;
1550 					particleContact.contactNormal = normal;
1551 					particleContact.penetration = distance - this.collisionRadius;
1552 					contacts.push(particleContact);
1553 					usedContacts++;
1554 				} // if
1555 			} // for
1556 		} // for
1557 		return usedContacts;
1558 	}

1613 				break;
1614 			} // if
1615

1642 				} // if

1644 				var particleContact = new ParticleContact();
1645 				particleContact.particles[0] = particle;
1646 				particleContact.contactNormal = normal;
1647 				particleContact.penetration = depth;
1648 				particleContact.restitution = 0.5;
1649 				contacts.push(particleContact);

1652 		} // for
