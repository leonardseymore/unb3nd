1 /**
  2  * @class
  3  * @constructor
  4  * Rectangle helper object
  5  * @param x X-Coordinate of the rectangle
  6  * @param y Y-Coordinate of the rectangle
  7  * @param width Width of the rectangle 
  8  * @param height Height of the rectangle
  9  */
 10 function Rectangle(x, y, width, height) {
 11 	this.x = (x == undefined ? 0 : x);
 12 	this.y = (y == undefined ? 0 : y);
 13 	this.width = (width == undefined ? 0 : width);
 14 	this.height = (height == undefined ? 0 : height);;
 15 
 16 	/**
 17 	 * @method
 18 	 * Draw this rectangle
 19 	 */
 20 	this.draw = function() {
 21 		ctx.strokeRect(this.x, this.y, this.width, this.height);
 22 	}
 23 	
 24 	/**
 25 	 * @method
 26 	 * Tests if a point is inside this rect
 27 	 * @param Vector2 point The point to test
 28 	 * @return true If this point is inside the rect
 29 	 */
 30 	this.isPointInside = function(point) {
 31 		if (point.x >= this.x &&
 32 			point.x <= this.x + this.width &&
 33 			point.y >= this.y &&
 34 			point.y <= this.y + this.height
 35 			) {
 36 			return true;
 37 		} // if
 38 		
 39 		return false;
 40 	}
 41 	
 42 	/**
 43 	 * @method
 44 	 * Tests if a point is strictly inside this rect
 45 	 * @param Vector2 point The point to test
 46 	 * @return true If this point is inside the rect
 47 	 */
 48 	this.isPointInsideStrict = function(point) {
 49 		if (point.x > this.x &&
 50 			point.x < this.x + this.width &&
 51 			point.y > this.y &&
 52 			point.y < this.y + this.height
 53 			) {
 54 			return true;
 55 		} // if
 56 		
 57 		return false;
 58 	}
 59 	
 60 	/**
 61 	 * @method
 62 	 * Shrinks the rectangle by the given amount
 63 	 * @return Rectangle The shrunked version of this rectangle
 64 	 */
 65 	this.shrink = function(amount) {
 66 		var clone = this.clone();
 67 		clone.shrinkMutate(amount);
 68 		return clone;
 69 	}
 70 	
 71 	/**
 72 	 * @method
 73 	 * Shrinks the rectangle by the given amount
 74 	 * @return void
 75 	 */
 76 	this.shrinkMutate = function(amount) {
 77 		this.x += amount;
 78 		this.y += amount;
 79 		this.width -= amount * 2;
 80 		this.height -= amount * 2;
 81 	}
 82 	
 83 	/**
 84 	 * @method
 85 	 * Clones this rectangle
 86 	 * @return Rectangle A clone of this rectangle
 87 	 */
 88 	this.clone = function() {
 89 		return new Rectangle(this.x, this.y, this.width, this.height);
 90 	}
 91 }

 93 /**
 94  * @class
 95  * @constructor
 96  * 2-Dimensional vector
 97  *
 98  * Since we want to use clones in some arithmetic and simply modify a vector
 99  * in other situations. Some of the operations come in two flavors. e.g. @link{#add} and
100  * @link{#addMutate}.  In this example add creates a new vector and leaves the current 
101  * unchanged, whereas @link{#addMutate} modifies this vector and returns nothing.
102  *
103  * @param float x X-coordinate
104  * @param float y Y-coordinate
105  */
106 function Vector2(x, y) {
107 	/**
108 	 * @field float
109 	 * The X-coordinate
110 	 */
111 	this.x = (x == undefined ? 0 : x);
112 	
113 	/**
114 	 * @field float
115 	 * The Y-coordinate
116 	 */
117 	this.y = (y == undefined ? 0 : y);
118 	
119 	/**
120 	 * @method
121 	 * Add another vector to this vector
122 	 * @param Vector2 o The vector to add to this vector
123 	 * @return Vector2 The newly created vector
124 	 */
125 	this.add = function(o) {
126 		var newVector = this.clone();
127 		newVector.x += o.x;
128 		newVector.y += o.y;
129 		return newVector;
130 	}
131 
132 	/**
133 	 * @method
134 	 * Add another vector to this vector
135 	 * @param Vector2 o The vector to add to this vector
136 	 * @return void
137 	 */
138 	this.addMutate = function(o) {
139 		this.x += o.x;
140 		this.y += o.y;
141 	}
142 	
143 	/**
144 	 * @method
145 	 * Remove another vector from this vector
146 	 * @param Vector2 o The vector to remove to this vector
147 	 * @return Vector2 The newly created vector
148 	 */
149 	this.sub = function(o) {
150 		var newVector = this.clone();
151 		newVector.x -= o.x;
152 		newVector.y -= o.y;
153 		return newVector;
154 	}
155 	
156 	/**
157 	 * @method
158 	 * Substract another vector from this vector
159 	 * @param Vector2 o The vector to subtract from this vector
160 	 * @return void
161 	 */
162 	this.subMutate = function(o) {
163 		this.x -= o.x;
164 		this.y -= o.y;
165 	}
166 	
167 	/**
168 	 * @method
169 	 * Flip vector around
170 	 * TODO: invert is maybe a better name for this method
171 	 * @return The flipped vector
172 	 */
173 	this.reverse = function() {
174 		return new Vector2(-this.x, -this.y);
175 	}
176 	
177 	/**
178 	 * @method
179 	 * Multiply by a scalar value
180 	 * @param float s Scalar to multiply by
181 	 * @return Vector2 The newly created vector
182 	 */
183 	this.multScalar = function(s) {
184 		var newVector = this.clone();
185 		newVector.x *= s;
186 		newVector.y *= s;
187 		return newVector;
188 	}
189 	
190 	/**
191 	 * @method
192 	 * Multiply by a scalar value
193 	 * @param float s Scalar to multiply by
194 	 * @return void
195 	 */
196 	this.multScalarMutate = function(s) {
197 		this.x *= s;
198 		this.y *= s;
199 	}
200 	
201 	/**
202 	 * @method
203 	 * Calculate the dot product with another vector
204 	 * @param Vector2 other The vector to perform the calculation with
205 	 * @return float The dot product
206 	 */
207 	this.dotProduct = function(other) {
208 		return this.x * other.x + this.y * other.y;
209 	}
210 	
211 	/**
212 	 * @method
213 	 * Calculate the magnitude of this vector
214 	 * @return float The magnitude
215 	 */
216 	this.getMagnitude = function() {
217 		return Math.sqrt(this.getMagnitudeSquare());
218 	}
219 	
220 	/**
221 	 * @method
222 	 * Calculate the squared magnitude of this vector
223 	 * @return float The square magnitude
224 	 */
225 	this.getMagnitudeSquare = function() {
226 		return this.x * this.x + this.y * this.y;
227 	}
228 	
229 	/**
230 	 * @method
231 	 * Normalizes this vector
232 	 * @return Vector2 The normalized form of this vector
233 	 */
234 	this.normalize = function() {
235 		var newVector = this.clone();
236 		newVector.normalizeMutate();
237 		return newVector;
238 	}
239 	
240 	/**
241 	 * @method
242 	 * Normalizes this vector
243 	 * @return void
244 	 */
245 	this.normalizeMutate = function() {
246 		var magnitude = this.getMagnitude();
247 		if (magnitude > 0) {
248 			this.multScalarMutate(1 / magnitude);
249 		} // if
250 	}
251 	
252 	/**
253 	 * @method
254 	 * Zero out this vector
255 	 * @return void
256 	 */
257 	this.zeroMutate = function() {
258 		this.x = 0;
259 		this.y = 0;
260 	}
261 	
262 	/**
263 	 * @method
264 	 * Clones this vector
265 	 * @return Vector2 The clone of this vector
266 	 */
267 	this.clone = function() {
268 		return new Vector2(this.x, this.y);
269 	}
270 	
271 	/**
272 	 * @method
273 	 * Converts the class to a string representation
274 	 * @return string The string representation of this class
275 	 */
276 	this.toString = function() {
277 		return "(" + this.x + ", " + this.y + ")";
278 	}
279 	
280 	/**
281 	 * @method
282 	 * Draw visual helpers for this vector
283 	 *
284 	 * This method assumes that translations have already been made to the starting
285 	 * position of the vector
286 	 * @param strokestyle Optional overriding stroke style
287 	 * @return void
288 	 */
289 	this.draw = function(strokestyle) {
290 		ctx.save();	
291 		
292 		ctx.strokeStyle = (strokestyle == undefined ? "lightblue" : strokestyle);
293 		ctx.beginPath();
294 		ctx.moveTo(0, 0);
295 		ctx.lineTo(this.x, this.y);
296 		ctx.stroke();
297 
298 		ctx.restore();
299 	}
300 	
301 	/**
302 	 * @method
303 	 * Draw visual helpers for this vector
304 	 *
305 	 * This method assumes that translations have already been made to the starting
306 	 * position of the vector
307 	 * @param fillStyle Optional overriding fill style
308 	 * @return void
309 	 */
310 	this.drawPoint = function(fillStyle) {
311 		ctx.save();	
312 		ctx.fillStyle = (fillStyle == undefined ? "magenta" : fillStyle);
313 		ctx.translate(this.x, this.y);
314 		ctx.fillRect(
315 			-1, -1, 2, 2
316 		);
317 		ctx.restore();
318 	}
319 }

321 /**
322  * @method
323  * @static
324  * Calculates the angle between two vectors in radians
325  * @param Vector2 v1 The first vector
326  * @param Vector2 v2 The second vector
327  * @return float The angle between the two vectors
328  */
329 Vector2.getAngle = function(v1, v2) {
330 	return Math.acos(v1.normalize().dotProduct(v2.normalize()));
331 }
