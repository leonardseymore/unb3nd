  7 /**
  8  * @class A particle is an object which has a position but no orientation
  9  * @constructor
 10  * @extends Observable
 11  * @param {float} mass Optional mass
 12  * @since 0.0.0
 13  */
 14 function Particle(mass) {
 15 
 16 	/*
 17 	 * Super constructor
 18 	 */
 19 	Observable.call(this);
 20 	
 21 	/**
 22 	 * Unique identifier to help track a single particle
 23 	 * @field 
 24 	 * @type long
 25 	 * @default Auto number
 26 	 * @since 0.0.0
 27 	 * @see Particle#getNextUid 
 28 	 */
 29 	this.uid = Particle.getNextUid();
 30 
 31 	/**
 32 	 * The position of this particle
 33 	 * @field 
 34 	 * @type Vector2
 35 	 * @default Zero vector
 36 	 * @since 0.0.0
 37 	 */
 38 	this.pos = new Vector2();
 39 	
 40 	/**
 41 	 * The velocity of this particle
 42 	 * @field
 43 	 * @type Vector2 
 44 	 * @default Zero vector
 45 	 * @since 0.0.0
 46 	 */
 47 	this.vel = new Vector2();
 48 	
 49 	/**
 50 	 * The acceleration of this particle
 51 	 * @field
 52 	 * @type Vector2 
 53 	 * @default Zero vector
 54 	 * @since 0.0.0
 55 	 */
 56 	this.acc = new Vector2();
 57 	
 58 	/**
 59 	 * Damping is a simple yet special property involved in slowing down moving objects 0 - 1, where 0 is full damping and 1 is no damping
 60 	 * @field
 61 	 * @type float 
 62 	 * @default 1.0
 63 	 * @since 0.0.0
 64 	 */
 65 	this.damping = 1.0;
 66 	
 67 	/**
 68 	 * Holds the inverse mass of the particle.
 69 	 * @private
 70 	 * @field
 71 	 * @type float 
 72 	 * @default 0, indicating infinite mass
 73 	 * @since 0.0.0
 74 	 */
 75 	this.inverseMass = 0;
 76 	if (mass) {
 77 		this.inverseMass = 1 / mass;
 78 	} // if
 79 	
 80 	/**
 81 	 * The overall force accumulator
 82 	 * @private
 83 	 * @field
 84 	 * @type Vector2 
 85 	 * @default Zero vector
 86 	 * @since 0.0.0
 87 	 */
 88 	this.forceAccum = new Vector2();
 89 	
 90 	/**
 91 	 * Determines if point is near this particle
 92 	 * @function 
 93 	 * @param {Vector2} point Point to test
 94 	 * @param {float} distance Distance to test
 95 	 * @returns {boolean} true if supplied point is near this particle
 96 	 * @since 0.0.0
 97 	 */
 98 	 this.isCloseToPoint = function(point, distance) {
 99     return Vector2.isWithin(this.pos, point, distance);
100 	 }
101 	
102 	/**
103 	 * Ensures this body has a finite mass
104 	 * @function 
105 	 * @returns {boolean} true if this body has a finite mass
106 	 * @since 0.0.0
107 	 */
108 	 this.hasFiniteMass = function() {
109 		return this.inverseMass > 0;
110 	 }
111 	
112 	/**
113 	 * Adds an external force to this particle
114 	 * @function 
115 	 * @param {Vector2} force The force to add to this particle
116 	 * @returns {void}
117 	 * @since 0.0.0
118 	 */
119 	this.applyForce = function(force) {
120 		this.forceAccum.addMutate(force);
121 	}
122 	
123 	/**
124 	 * Reset the force accumulator on this particle
125 	 * @function 
126 	 * @returns {void}
127 	 * @since 0.0.0
128 	 */
129 	this.clearForceAccum = function() {
130 		this.forceAccum.zeroMutate();
131 	}
132 	
133 	/**
134 	 * Sets the inverse mass of the particle
135 	 * @function 
136 	 * @param {float} inverseMass The inverse mass of the particle
137 	 * @returns {void}
138 	 * @since 0.0.0
139 	 */
140 	this.setInverseMass = function(inverseMass) {
141 		this.inverseMass = inverseMass;
142 	}
143 	
144 	/**
145 	 * Sets the mass of the particle
146 	 * @function
147 	 * @param {float} mass The mass of the particle
148 	 * @returns {void}
149 	 * @since 0.0.0
150 	 */
151 	this.setMass = function(mass) {
152 		if (mass == 0) {
153 			this.inverseMass = 0;
154 		} else {
155 			this.inverseMass = 1 / mass;
156 		} // if
157 	}
158 	
159 	/**
160 	 * Gets the mass of the particle
161 	 * @function
162 	 * @returns {float} mass The mass of the particle
163 	 * @since 0.0.0
164 	 */
165 	this.getMass = function() {
166 		if (this.inverseMass == 0) {
167 			return 0;
168 		} else {
169 			return 1 / this.inverseMass;
170 		} // if
171 	}
172 	
173 	/**
174 	 * Integrates the particle forward in time by the given amount.
175 	 * Uses Newton-Euler integration function
176 	 * @function
177 	 * @param {int} delta The time delta in milliseconds
178 	 * @returns {void}
179 	 * @since 0.0.0
180 	 */
181 	this.integrate = function(delta) {		
182 		var dt = delta / 1000;
183 		
184 		this.pos.addMutate(
185 			this.vel.multScalar(dt)
186 		);
187 
188 		var resultingAcc = this.acc.add(
189 			this.forceAccum.multScalar(this.inverseMass)
190 		);
191 		
192 		this.vel.addMutate(
193 			resultingAcc.multScalar(dt)
194 		);
195 		
196 		this.vel.multScalarMutate(
197 			Math.pow(this.damping, dt)
198 		);
199 		
200 		this.clearForceAccum();
201 	}
202 	
203 	/**
204 	 * On die callback
205 	 * @function
206 	 * @returns {void}
207 	 * @since 0.0.0
208 	 */
209 	this.ondie = undefined;
210 
211 	/**
212 	 * Die callback invoker
213 	 * @function
214 	 * @returns {void}
215 	 * @since 0.0.0
216 	 */
217 	this.die = function() {		
218 		if (this.ondie) {
219 			this.ondie();
220 		} // if
221 		this.dispatchEvent("die");
222 	}
223 
224   /**
225    * Accepts a particle world visitor
226 	 * @function
227    * @param {ParticleWorldVisitor} visitor Visitor to visit
228 	 * @returns {void}
229 	 * @since 0.0.0.3
230    */
231   this.accept = function(visitor) {
232     visitor.visitParticle(this);
233   }
234 	
235 	/**
236 	 * Converts the class to a string representation
237 	 * @function
238 	 * @returns {string} The string representation of this class
239 	 * @since 0.0.0
240 	 */
241 	this.toString = function() {
242 		return "uid=" + this.uid;
243 	}
244 }

270 /** 
271  * @class A force registry for matching up force generators to particles
272  * @constructor
273  * @since 0.0.0
274  */
275 function ParticleForceRegistry() {
276 	
277 	/**
278 	 * Registry entries
279 	 * @field
280 	 * @type {"particle":Particle, "forceGenerators":ParticleForceGenerator []}
281 	 * @default []
282 	 * @since 0.0.0
283 	 */
284 	this.entries = [];
285 	
286 	/**
287 	 * Adds the force generator to the supplied particle
288 	 * TODO: This is a very brute force function, we need a HashMap or similar datastructure
289 	 * @function
290 	 * @param {Particle} particle The particle to add a force generator to
291 	 * @param {ParticleForceGenerator} forceGenerator The force generator to add to the particle
292 	 * @returns {void}
293 	 * @since 0.0.0
294 	 */
295 	this.add = function(particle, forceGenerator) {
296 		var entry;
297 		var found = false;
298 		for (i in this.entries) {
299 			entry = this.entries[i];
300 			if (entry.particle === particle) {
301 				found = true;
302 				break;
303 			} // if
304 		} // for
305 		
306 		if (!found) {
307 			entry = {"particle":particle, "forceGenerators":new Array(forceGenerator)};
308 			this.entries.push(entry);
309 		} else {
310 			entry.forceGenerators.push(forceGenerator);
311 		} // if
312 	}
313 	
314 	/**
315 	 * Gets all force generators for the supplied particle
316 	 * @function
317 	 * @param {Particle} particle The particle to get the force generators for
318 	 * @returns {ParticleForceGenerator []} All the force generators for the supplied particle, undefined if not found
319 	 * @since 0.0.0
320 	 */
321 	this.getForceGenerators = function(particle) {
322 		for (i in this.entries) {
323 			var entry = this.entries[i];
324 			if (entry.particle === particle) {
325 				return entry.forceGenerators;
326 			} // if
327 		} // for
328 		
329 		return undefined;
330 	}
331 	
332 	/**
333 	 * Removes all force generators for the supplied particle
334 	 * @function
335 	 * @param {Particle} particle The particle to remove the force generators from
336 	 * @returns {void}
337 	 * @since 0.0.0
338 	 */
339 	this.removeForceGenerators = function(particle) {
340 		for (i in this.entries) {
341 			var entry = this.entries[i];
342 			if (entry.particle === particle) {
343 				this.entries.splice(i, 1);
344 			} // if
345 		} // for
346 	}
347 	
348 	/**
349 	 * Removes a specific force generator
350 	 * @function
351 	 * @param {ParticleForceGenerator} forceGenerator The specific force generator to remove
352 	 * @return {void}
353 	 * @since 0.0.0
354 	 */
355 	this.removeForceGenerator = function(forceGenerator) {
356 		for (i in this.entries) {
357 			var entry = this.entries[i];
358 			for (j in entry.forceGenerators) {
359 				var fg = entry.forceGenerators[j];
360 				if (fg === forceGenerator) {
361 					entry.forceGenerators.splice(j, 1);
362 				} // if
363 			} // if
364 		} // for
365 	}
366 	
367 	/**
368 	 * Adds each force generator's force to every particle in the registry over the delta time
369 	 * @function
370 	 * @param {int} delta Delta time in milliseconds
371 	 * @returns {void}
372 	 * @since 0.0.0
373 	 */
374 	this.applyForces = function(delta) {
375 		for (i in this.entries) {
376 			var entry = this.entries[i];
377 			var particle = entry.particle;
378 			var forceGenerators = entry.forceGenerators;
379 			for (j in forceGenerators) {
380 				var forceGenerator = forceGenerators[j];
381 				forceGenerator.applyForce(particle, delta);
382 			} // for
383 		} // for
384 	}
385 
386   /**
387    * Accepts a particle world visitor
388 	 * @method
389    * @abstract
390    * @param {ParticleWorldVisitor} visitor Visitor to visit
391 	 * @returns {void}
392 	 * @since 0.0.0.3
393    */
394 	this.accept = function(visitor) {
395     for (i in this.entries) {
396 			var entry = this.entries[i];
397 			var particle = entry.particle;
398 			var forceGenerators = entry.forceGenerators;
399 			for (j in forceGenerators) {
400 				var forceGenerator = forceGenerators[j];
401 				forceGenerator.accept(visitor, particle);
402 			} // for
403 		} // for
404 	}
405 }

573 /**
574  * @class A drag force generator
575  * @constructor
576  * @extends ParticleForceGenerator
577  * @param {float} k1 Velocity drag coefficient
578  * @param {float} k2 Velocity squared drag coefficient
579  * @since 0.0.0
580  */
581 function ParticleDragForceGenerator(k1, k2) {
582 
583 	/**
584 	 * Velocity drag coefficient
585 	 * @field
586 	 * @type float
587 	 * @default Defaults to {@link DEFAULT_DRAG_VELOCITY_COEFF}
588 	 * @since 0.0.0
589 	 */
590 	this.k1 = k1 || DEFAULT_DRAG_VELOCITY_COEFF;
591 	
592 	/**
593 	 * Velocity squared drag coefficient
594 	 * @field 
595 	 * @type float
596 	 * @default Defaults to {@link DEFAULT_DRAG_VELOCITY_SQUARED_COEFF}
597 	 * @since 0.0.0
598 	 */
599 	this.k2 = k2 || DEFAULT_DRAG_VELOCITY_SQUARED_COEFF;
600 	
601 	/**
602 	 * Apply gravity to the given mass over the delta time
603 	 * @function
604 	 * @override
605 	 * @param {Particle} particle The particle to apply the force to
606 	 * @param {int} delta The delta time in milliseconds
607 	 * @returns {void}
608 	 * @since 0.0.0
609 	 */
610 	this.applyForce = function(particle, delta) {
611 		var force = this.calculateForce(particle, delta);
612 		particle.applyForce(force);
613 	}
614 	
615 	/**
616 	 * Calculate the force for the supplied particle over the time delta
617 	 * @private
618 	 * @function
619 	 * @param {Particle} particle The particle to apply the force to
620 	 * @param {int} delta The delta time in milliseconds
621 	 * @returns {Vector2} The force to apply to this particle
622 	 * @since 0.0.0
623 	 */
624 	this.calculateForce = function(particle, delta) {
625 		var dragCoeff = particle.vel.getMagnitude();
626 		dragCoeff = this.k1 * dragCoeff + this.k2 * dragCoeff * dragCoeff;
627 		
628 		var force = particle.vel.normalize();
629 		force.multScalarMutate(-dragCoeff);
630 		return force;
631 	}
632 
633   /**
634 	 * Accepts the supplied particle visitor
635 	 * @function
636 	 * @override
637 	 */
638 	this.accept = function(visitor, particle) {
639     visitor.visitDragForceGenerator(this, particle);
640 	}
641 	
642 	/**
643 	 * Converts the class to a string representation
644 	 * @function
645 	 * @return {string} The string representation of this class
646 	 * @since 0.0.0
647 	 */
648 	this.toString = function() {
649 		return "Drag: k1=" + this.k1 + " ,k2=" + this.k2;
650 	}

654 /**
655  * @class A spring force generator
656  * @constructor
657  * @extends ParticleForceGenerator
658  * @param {Particle} particleOther Particle at the other end of the spring
659  * @param {float} springConstant Holds the spring constant
660  * @param {float} restLength Holds the spring's rest length
661  * @since 0.0.0
662  */
663 function ParticleSpringForceGenerator(particleOther, springConstant, restLength) {
664 
665 	/**
666 	 * Particle at the other end of the spring
667 	 * @field 
668 	 * @type Particle
669 	 * @default particleOther
670 	 * @since 0.0.0
671 	 */
672 	this.particleOther = particleOther;
673 	
674 	/**
675 	 * Holds the spring constant
676 	 * @field
677 	 * @type float
678 	 * @default springConstant
679 	 * @since 0.0.0
680 	 */
681 	this.springConstant = springConstant;
682 	
683 	/**
684 	 * Holds the spring's rest length
685 	 * @field
686 	 * @type float
687 	 * @default restLength
688 	 * @since 0.0.0
689 	 */
690 	this.restLength = restLength;
691 	
692 	/**
693 	 * Apply spring force to the given particle over the delta time
694 	 * @function
695 	 * @override
696 	 * @param {Particle} particle The particle to apply the force to
697 	 * @param {int} delta The delta time in milliseconds
698 	 * @returns {void}
699 	 * @since 0.0.0
700 	 */
701 	this.applyForce = function(particle, delta) {
702 		var force = particle.pos.clone();
703 		force.subMutate(this.particleOther.pos);
704 		
705 		var magnitude = force.getMagnitude();
706 		magnitude = Math.abs(magnitude - this.restLength);
707 		magnitude *= this.springConstant;
708 		
709 		force.normalizeMutate();
710 		force.multScalarMutate(-magnitude);
711 		particle.applyForce(force);
712 	}
713

715 	 * Accepts the supplied particle visitor
716 	 * @function
717 	 * @override
718 	 */
719 	this.accept = function(visitor, particle) {

721 	}

1217 /**
1218  * @class A contact represents two objects in contact (in this case
1219  * ParticleContact representing two particles). Resolving a
1220  * contact removes their interpenetration, and applies sufficient
1221  * impulse to keep them apart. Colliding bodies may also rebound.
1222  *
1223  * The contact has no callable functions, it just holds the
1224  * contact details. To resolve a set of contacts, use the particle
1225  * contact resolver class.
1226  * @constructor
1227  * @since 0.0.0
1228  */
1229 function ParticleContact() {
1230 
1231 	/**
1232 	 * Holds the particles that are involved in the contact. The second may be
1233 	 * omitted in the case of a contact with the environment.
1234 	 * @field
1235 	 * @type Particle []
1236 	 * @default []
1237 	 * @since 0.0.0
1238 	 */
1239 	this.particles = [];
1240 	
1241 	/**
1242 	 * Holds the normal restitution coefficient at the contact
1243 	 * @field 
1244 	 * @type float
1245 	 * @default 0.0
1246 	 * @since 0.0.0
1247 	 */
1248 	this.restitution = 0.0;
1249 	
1250 	/**
1251 	 * Holds the contact normal
1252 	 * @field 
1253 	 * @type Vector2
1254 	 * @default Zero vector
1255 	 * @since 0.0.0
1256 	 */
1257 	this.contactNormal = new Vector2();
1258 	
1259 	/**
1260 	 * Penetration depth in the direction of the contact normal
1261 	 * @field 
1262 	 * @type float
1263 	 * @default 0.0
1264 	 * @since 0.0.0
1265 	 */
1266 	this.penetration = 0.0;
1267 	
1268 	/**
1269 	 * Resolves this contact for both veolcity and interpenetration
1270 	 * @function
1271 	 * @protected
1272 	 * @param {int} delta Delta time in milliseconds since last update
1273 	 * @returns {void}
1274 	 * @since 0.0.0
1275 	 */
1276 	this.resolve = function(delta) {
1277 		this.resolveVelocity(delta);
1278 		this.resolveInterpenetration(delta);
1279 	}
1280 	
1281 	/**
1282 	 * Calculate separating velocity at this contact
1283 	 * @function
1284 	 * @protected
1285 	 * @returns {float} Separating velocity at this contact
1286 	 * @since 0.0.0
1287 	 */
1288 	this.calculateSeparatingVelocity = function() {
1289 		var relativeVelocity = this.particles[0].vel.clone();
1290 		if (this.particles[1]) {
1291 			relativeVelocity.subMutate(
1292 				this.particles[1].vel
1293 			);
1294 		} // if
1295 		return relativeVelocity.dotProduct(this.contactNormal);
1296 	}
1297 	
1298 	/**
1299 	 * Handles impulse calculations for this collision
1300 	 * @function
1301 	 * @private
1302 	 * @param {int} delta Delta time in milliseconds since last update
1303 	 * @returns {void}
1304 	 * @since 0.0.0
1305 	 */
1306 	this.resolveVelocity = function(delta) {
1307 		var dt = delta / 1000;
1308 		var separatingVelocity = this.calculateSeparatingVelocity();
1309 		if (separatingVelocity > 0) {
1310 			// contact is either separating or stationary
1311 			return;
1312 		} // if
1313 		var newSepVelocity = -separatingVelocity * this.restitution;
1314 		
1315 		// ~PHYLIMIT keep to objects in resting contact
1316 		// micro-collision approach
1317 		// check velocity build up due to acceleration only
1318 		var accCausedVelocity = this.particles[0].acc.clone();
1319 		if (this.particles[1]) {
1320 			accCausedVelocity.subMutate(this.particles[1].acc);
1321 		} // if
1322 		var accCausedSepVelocity = accCausedVelocity.dotProduct(this.contactNormal) * dt;
1323 		
1324 		// If we've got a closing velocity due to acceleration build-up,
1325 		// remove it from the new separating velocity.
1326 		if (accCausedSepVelocity < 0) {
1327 			newSepVelocity += resitution * accCausedSepVelocity;
1328 			
1329 			// Make sure we haven't removed more than was there to remove.
1330 			if (newSepVelocity < 0) {
1331 				newSepVelocity = 0;
1332 			} // if
1333 		} // if
1334 		// ~END-PHYLIMIT
1335 		
1336 		var deltaVelocity = newSepVelocity - separatingVelocity;
1337 		
1338 		var totalInverseMass = this.particles[0].inverseMass;
1339 		if (this.particles[1]) {
1340 			totalInverseMass += this.particles[1].inverseMass;
1341 		} // if
1342 		
1343 		if (totalInverseMass <= 0) {
1344 			// all particles have infinite mass
1345 			return;
1346 		} // if
1347 		
1348 		var impulse = deltaVelocity / totalInverseMass;
1349 		var impulsePerIMass = this.contactNormal.multScalar(impulse);
1350 		
1351 		// apply impulses proportional to inverse mass
1352 		// in the direction of the contact
1353 		this.particles[0].vel.addMutate(
1354 			impulsePerIMass.multScalar(this.particles[0].inverseMass)
1355 		);
1356 		
1357 		if (this.particles[1]) {
1358 			this.particles[1].vel.addMutate(
1359 				impulsePerIMass.multScalar(-this.particles[1].inverseMass)
1360 			);
1361 		} // if
1362 	}
1363 	
1364 	/**
1365 	 * Handles interpenetration resolution for this contact
1366 	 * @function
1367 	 * @param {int} delta Delta time in milliseconds since last update
1368 	 * @returns {void}
1369 	 * @since 0.0.0
1370 	 */
1371 	this.resolveInterpenetration = function(delta) {
1372 		if (this.penetration <= 0) {
1373 			return;
1374 		} // if
1375 		
1376 		var totalInverseMass = this.particles[0].inverseMass;
1377 		if (this.particles[1]) {
1378 			totalInverseMass += this.particles[1].inverseMass;
1379 		} // if
1380 		
1381 		if (totalInverseMass <= 0) {
1382 			// infinite mass
1383 			return;
1384 		} // if
1385 		
1386 		// find penetration resolution relative to inverse mass
1387 		var movePerIMass = this.contactNormal.multScalar(
1388 			-this.penetration / totalInverseMass
1389 		);
1390 		
1391 		this.particles[0].pos.addMutate(
1392 			movePerIMass.multScalar(this.particles[0].inverseMass)
1393 		);
1394 		
1395 		if (this.particles[1]) {
1396 			this.particles[1].pos.addMutate(
1397 				movePerIMass.multScalar(this.particles[1].inverseMass)
1398 			);
1399 		} // if
1400 	}
1401 }

1403 /**
1404  * @class The contact resolution routine for particle contacts. One resolver
1405  * instance can be shared for the whole simulation.
1406  * @constructor
1407  * @since 0.0.0
1408  */
1409 function ParticleContactResolver() {
1410 	/**
1411 	 * Number of allowable iterations
1412 	 * Recommend: Minimally number of contacts, as a single contact is resolved
1413 	 *            per iteration.
1414 	 * @field 
1415 	 * @type int
1416 	 * @default 0
1417 	 * @since 0.0.0
1418 	 */
1419 	this.maxIt = 0;
1420 
1421 	/**
1422 	 * Resolves the set of particle contacts for both penetration and velocity
1423 	 * @function
1424 	 * @param {ParticleContact []} contacts All contacts to resolve
1425 	 * @param {int} delta Delta time in milliseconds since last update
1426 	 * @returns {void}
1427 	 * @since 0.0.0
1428 	 */
1429 	this.resolveContacts = function(contacts, delta) {
1430 		var it = 0;
1431 		while (it < this.maxIt) {
1432 			// find the contact with the largest closing velocity
1433 			var max = 0;
1434 			var maxContact = contacts[0];
1435 			for (i in contacts) {
1436 				var contact = contacts[i];
1437 				var sepVel = contact.calculateSeparatingVelocity();
1438 				if (sepVel < max) {
1439 					max = sepVel;
1440 					maxContact = contact;
1441 				} // if
1442 			} // for
1443 			maxContact.resolve(delta);
1444 			it++;
1445 		} // while
1446 	}
1447 }
