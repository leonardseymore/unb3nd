  7 /**
  8  * @class Rectangle helper object
  9  * @constructor
 10  * @param {float} x X-Coordinate of the rectangle
 11  * @param {float} y Y-Coordinate of the rectangle
 12  * @param {float} width Width of the rectangle 
 13  * @param {float} height Height of the rectangle
 14  * @since 0.0.0
 15  */
 16 function Rectangle(x, y, width, height) {
 17 
 18 	/**
 19 	 * X position of this rectangle
 20 	 * @field 
 21 	 * @type float
 22 	 * @default 0
 23 	 * @since 0.0.0
 24 	 */
 25 	this.x = x || 0;
 26 	
 27 	/**
 28 	 * Y position of this rectangle
 29 	 * @field 
 30 	 * @type float
 31 	 * @default 0
 32 	 * @since 0.0.0
 33 	 */
 34 	this.y = y || 0;
 35 	
 36 	/**
 37 	 * Width of this rectangle
 38 	 * @field 
 39 	 * @type float
 40 	 * @default 0
 41 	 * @since 0.0.0
 42 	 */
 43 	this.width = width || 0;
 44 	
 45 	/**
 46 	 * Height of this rectangle
 47 	 * @field 
 48 	 * @type float
 49 	 * @default 0
 50 	 * @since 0.0.0
 51 	 */
 52 	this.height = height || 0;
 53 
 54 	/**
 55 	 * Draws this rectangle
 56 	 * @function 
 57 	 * @returns {void}
 58 	 * @since 0.0.0
 59 	 */
 60 	this.draw = function() {
 61 		ctx.strokeRect(this.x, this.y, this.width, this.height);
 62 	}
 63 	
 64 	/**
 65 	 * Tests if a point is inside this rect
 66 	 * @function
 67 	 * @param {Vector2} point The point to test
 68 	 * @returns {boolean} true if this point is inside the rect
 69 	 * @since 0.0.0
 70 	 */
 71 	this.isPointInside = function(point) {
 72 		if (point.x >= this.x &&
 73 			point.x <= this.x + this.width &&
 74 			point.y >= this.y &&
 75 			point.y <= this.y + this.height
 76 			) {
 77 			return true;
 78 		} // if
 79 		
 80 		return false;
 81 	}
 82 	
 83 	/**
 84 	 * Tests if a point is strictly inside this rect
 85 	 * @function
 86 	 * @param {Vector2} point The point to test
 87 	 * @returns {boolean} true if this point is inside the rect
 88 	 * @since 0.0.0
 89 	 */
 90 	this.isPointInsideStrict = function(point) {
 91 		if (point.x > this.x &&
 92 			point.x < this.x + this.width &&
 93 			point.y > this.y &&
 94 			point.y < this.y + this.height
 95 			) {
 96 			return true;
 97 		} // if
 98 		
 99 		return false;
100 	}
101 	
102 	/**
103 	 * Shrinks the rectangle by the given amount
104 	 * @function
105 	 * @param {float} amount The amount by which to shrink this rectangle
106 	 * @return {Rectangle} The shrunked version of this rectangle
107 	 * @since 0.0.0
108 	 */
109 	this.shrink = function(amount) {
110 		var clone = this.clone();
111 		clone.shrinkMutate(amount);
112 		return clone;
113 	}
114 	
115 	/**
116 	 * Shrinks the rectangle by the given amount
117 	 * @function
118 	 * @param {float} amount The amount by which to shrink this rectangle
119 	 * @return {void}
120 	 * @since 0.0.0
121 	 */
122 	this.shrinkMutate = function(amount) {
123 		this.x += amount;
124 		this.y += amount;
125 		this.width -= amount * 2;
126 		this.height -= amount * 2;
127 	}
128 	
129 	/**
130 	 * Clones this rectangle
131 	 * @function
132 	 * @return {Rectangle} The cloned version of this rectangle
133 	 * @since 0.0.0
134 	 */
135 	this.clone = function() {
136 		return new Rectangle(this.x, this.y, this.width, this.height);
137 	}
138 
139   /**
140 	 * Converts the class to a string representation
141 	 * @function
142 	 * @returns {string} The string representation of this class
143 	 * @since 0.0.0.3
144 	 */
145 	this.toString = function() {
146 		return "(x=" + this.x + ", y=" + this.y + ", w=" + this.width + ", h=" + this.height + ")";
147 	}
148 }

150 /**
151  * @class 2-Dimensional vector
152  *
153  * Since we want to use clones in some arithmetic and simply modify a vector

200 	 * @returns {void}
201 	 * @since 0.0.0
202 	 */

422 /**
423  * Calculates the angle between two vectors in radians
424  * @function
425  * @static
426  * @param {Vector2} v1 The first vector
427  * @param {Vector2} v2 The second vector
428  * @returns {float} The angle between the two vectors
429  * @since 0.0.0
430  */
431 Vector2.getAngle = function(v1, v2) {
432 	return Math.acos(v1.normalize().dotProduct(v2.normalize()));
433 }

435 /**
436  * Calculates the distance between two vectors
437  * @function
438  * @static
439  * @param {Vector2} v1 The first vector
440  * @param {Vector2} v2 The second vector
441  * @returns {float} The distance between the two vectors
442  * @since 0.0.0.3
443  */
444 Vector2.getDistance = function(v1, v2) {
445 	return v1.sub(v2).getMagnitude();
446 }
447 
448 /**
449  * Calculates the squared distance between two vectors
450  * This method is less expensive than {@link Vector2#getDistance} method
451  * because it does not perform a square root operation.
452  * @function
453  * @static
454  * @param {Vector2} v1 The first vector
455  * @param {Vector2} v2 The second vector
456  * @returns {float} The distance squared between the two vectors
457  * @since 0.0.0.3
458  */
459 Vector2.getDistanceSquare = function(v1, v2) {
460 	return v1.sub(v2).getMagnitudeSquare();
461 }
462 
463 /**
464  * Determines if the two vectors are within a specified distance from each other.
465  * @function
466  * @static
467  * @param {Vector2} v1 The first vector
468  * @param {Vector2} v2 The second vector
469  * @param {float} distance The distance to test for
470  * @returns {boolean} True if the vectors are within specified distance
471  * @since 0.0.0.3
472  */
473 Vector2.isWithin = function(v1, v2, distance) {
474 	return Vector2.getDistanceSquare(v1, v2) <= distance * distance;
475 }

717 /**
718  * @class 2x2 matrix
719  * @constructor
720  * @param {float} data1 Matrix data entry
721  * @param {float} data2 Matrix data entry
722  * @param {float} data3 Matrix data entry
723  * @param {float} data4 Matrix data entry
724  * @since 0.0.0
725  */
726 function Matrix2(data1, data2, data3, data4, inverse) {
727 
728 	/**
729 	 * Matrix entries
730 	 * @field
731 	 * @type float []
732 	 * @default [0.0, 0.0, 0.0, 0.0]
733 	 * @since 0.0.0
734 	 */
735 	this.e = [
736 		data1 || 0.0, data2 || 0.0, data3 || 0.0, data4 || 0.0
737 	];
738 	
739 	/**
740 	 * Matrix determinant
741 	 * @field
742 	 * @type float
743 	 * @default 0.0
744 	 * @since 0.0.0
745 	 */
746 	this.det = 0.0;
747 	
748 	/**
749 	 * Matrix inverse
750 	 * @field
751 	 * @type Matrix2
752 	 * @default undefined
753 	 * @since 0.0.0
754 	 */
755 	this.inverse = undefined;
756 	
757 	/**
758 	 * Calculates this determinant {@link Matrix2#det} from matrix entries {@link Matrix2#e}
759 	 * @function
760 	 * @private
761 	 * @returns {void}
762 	 * @since 0.0.0
763 	 */
764 	this.recaclulateDeterminant = function() {
765 		this.det = this.e[0] * this.e[3] - this.e[1] * this.e[2];
766 	}
767 	this.recaclulateDeterminant();
768 	
769 	/**
770 	 * Calculates this inverse {@link Matrix2#inverse} from matrix entries {@link Matrix2#e}
771 	 * @function
772 	 * @private
773 	 * @returns {void}
774 	 * @since 0.0.0
775 	 */
776 	this.recaclulateInverse = function() {
777 		
778 		if (this.det == 0) {
779 			return;
780 		} // if
781 		
782 		var inverseDet = 1 / this.det;
783 		var i = new Matrix2(
784 			inverseDet * this.e[3],
785 			-inverseDet * this.e[1],
786 			-inverseDet * this.e[2],
787 			inverseDet * this.e[0]
