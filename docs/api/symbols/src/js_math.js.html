  7 /**
  8  * @class Rectangle helper object
  9  * @constructor
 10  * @param {float} x X-Coordinate of the rectangle
 11  * @param {float} y Y-Coordinate of the rectangle
 12  * @param {float} width Width of the rectangle 
 13  * @param {float} height Height of the rectangle
 14  * @since 0.0.0
 15  */
 16 function Rectangle(x, y, width, height) {
 17 
 18 	/**
 19 	 * X position of this rectangle
 20 	 * @field 
 21 	 * @type float
 22 	 * @default 0
 23 	 * @since 0.0.0
 24 	 */
 25 	this.x = x || 0;
 26 	
 27 	/**
 28 	 * Y position of this rectangle
 29 	 * @field 
 30 	 * @type float
 31 	 * @default 0
 32 	 * @since 0.0.0
 33 	 */
 34 	this.y = y || 0;
 35 	
 36 	/**
 37 	 * Width of this rectangle
 38 	 * @field 
 39 	 * @type float
 40 	 * @default 0
 41 	 * @since 0.0.0
 42 	 */
 43 	this.width = width || 0;
 44 	
 45 	/**
 46 	 * Height of this rectangle
 47 	 * @field 
 48 	 * @type float
 49 	 * @default 0
 50 	 * @since 0.0.0
 51 	 */
 52 	this.height = height || 0;
 53 
 54 	/**
 55 	 * Draws this rectangle
 56 	 * @function 
 57 	 * @returns {void}
 58 	 * @since 0.0.0
 59 	 */
 60 	this.draw = function() {
 61 		ctx.strokeRect(this.x, this.y, this.width, this.height);
 62 	}
 63 	
 64 	/**
 65 	 * Tests if a point is inside this rect
 66 	 * @function
 67 	 * @param {Vector2} point The point to test
 68 	 * @returns {boolean} true if this point is inside the rect
 69 	 * @since 0.0.0
 70 	 */
 71 	this.isPointInside = function(point) {
 72 		if (point.x >= this.x &&
 73 			point.x <= this.x + this.width &&
 74 			point.y >= this.y &&
 75 			point.y <= this.y + this.height
 76 			) {
 77 			return true;
 78 		} // if
 79 		
 80 		return false;
 81 	}
 82 	
 83 	/**
 84 	 * Tests if a point is strictly inside this rect
 85 	 * @function
 86 	 * @param {Vector2} point The point to test
 87 	 * @returns {boolean} true if this point is inside the rect
 88 	 * @since 0.0.0
 89 	 */
 90 	this.isPointInsideStrict = function(point) {
 91 		if (point.x > this.x &&
 92 			point.x < this.x + this.width &&
 93 			point.y > this.y &&
 94 			point.y < this.y + this.height
 95 			) {
 96 			return true;
 97 		} // if
 98 		
 99 		return false;
100 	}
101 	
102 	/**
103 	 * Shrinks the rectangle by the given amount
104 	 * @function
105 	 * @param {float} amount The amount by which to shrink this rectangle
106 	 * @return {Rectangle} The shrunked version of this rectangle
107 	 * @since 0.0.0
108 	 */
109 	this.shrink = function(amount) {
110 		var clone = this.clone();
111 		clone.shrinkMutate(amount);
112 		return clone;
113 	}
114 	
115 	/**
116 	 * Shrinks the rectangle by the given amount
117 	 * @function
118 	 * @param {float} amount The amount by which to shrink this rectangle
119 	 * @return {void}
120 	 * @since 0.0.0
121 	 */
122 	this.shrinkMutate = function(amount) {
123 		this.x += amount;
124 		this.y += amount;
125 		this.width -= amount * 2;
126 		this.height -= amount * 2;
127 	}
128 	
129 	/**
130 	 * Clones this rectangle
131 	 * @function
132 	 * @return {Rectangle} The cloned version of this rectangle
133 	 * @since 0.0.0
134 	 */
135 	this.clone = function() {
136 		return new Rectangle(this.x, this.y, this.width, this.height);
137 	}
138 }

140 /**
141  * @class 2-Dimensional vector
142  *
143  * Since we want to use clones in some arithmetic and simply modify a vector
144  * in other situations. Some of the operations come in two flavors. e.g. @link{#add} and
145  * @link{#addMutate}.  In this example add creates a new vector and leaves the current 
146  * unchanged, whereas @link{#addMutate} modifies this vector and returns nothing.
147  * @constructor
148  * @param {float} x X-coordinate
149  * @param {float} y Y-coordinate
150  * @since 0.0.0
151  */
152 function Vector2(x, y) {
153 
154 	/**
155 	 * X coordinate of this rectangle
156 	 * @field 
157 	 * @type float
158 	 * @default 0
159 	 * @since 0.0.0
160 	 */
161 	this.x = x || 0;
162 	
163 	/**
164 	 * Y coordinate of this rectangle
165 	 * @field 
166 	 * @type float
167 	 * @default 0
168 	 * @since 0.0.0
169 	 */
170 	this.y = y || 0;
171 	
172 	/**
173 	 * Add another vector to this vector
174 	 * @function
175 	 * @param {Vector2} o The vector to add to this vector
176 	 * @return {Vector2} The newly created vector
177 	 * @since 0.0.0
178 	 */
179 	this.add = function(o) {
180 		var newVector = this.clone();
181 		newVector.x += o.x;
182 		newVector.y += o.y;
183 		return newVector;
184 	}
185 
186 	/**
187 	 * Add another vector to this vector
188 	 * @function
189 	 * @param {Vector2} o The vector to add to this vector
190 	 * @returns {void}
191 	 * @since 0.0.0
192 	 */
193 	this.addMutate = function(o) {
194 		this.x += o.x;
195 		this.y += o.y;
196 	}
197 	
198 	/**
199 	 * Remove another vector from this vector
200 	 * @function
201 	 * @param {Vector2} o The vector to remove to this vector
202 	 * @return {Vector2} The newly created vector
203 	 * @since 0.0.0
204 	 */
205 	this.sub = function(o) {
206 		var newVector = this.clone();
207 		newVector.x -= o.x;
208 		newVector.y -= o.y;
209 		return newVector;
210 	}
211 	
212 	/**
213 	 * Substract another vector from this vector
214 	 * @function
215 	 * @param {Vector2} o The vector to subtract from this vector
216 	 * @returns {void}
217 	 * @since 0.0.0
218 	 */
219 	this.subMutate = function(o) {
220 		this.x -= o.x;
221 		this.y -= o.y;
222 	}
223 	
224 	/**
225 	 * Flip vector around
226 	 * TODO: invert is maybe a better name for this method
227 	 * @function
228 	 * @returns {Vector2} The flipped vector
229 	 * @since 0.0.0
230 	 */
231 	this.reverse = function() {
232 		return new Vector2(-this.x, -this.y);
233 	}
234 	
235 	/**
236 	 * Multiply by a scalar value
237 	 * @function
238 	 * @param {float} s Scalar to multiply by
239 	 * @returns {Vector2} The newly created vector
240 	 * @since 0.0.0
241 	 */
242 	this.multScalar = function(s) {
243 		var newVector = this.clone();
244 		newVector.x *= s;
245 		newVector.y *= s;
246 		return newVector;
247 	}
248 	
249 	/**
250 	 * Multiply by a scalar value
251 	 * @function
252 	 * @param {float} s Scalar to multiply by
253 	 * @returns {void}
254 	 * @since 0.0.0
255 	 */
256 	this.multScalarMutate = function(s) {
257 		this.x *= s;
258 		this.y *= s;
259 	}
260 	
261 	/**
262 	 * Calculate the dot product with another vector
263 	 * @function
264 	 * @param {Vector2} other The vector to perform the calculation with
265 	 * @returns {float} The dot product
266 	 * @since 0.0.0
267 	 */
268 	this.dotProduct = function(other) {
269 		return this.x * other.x + this.y * other.y;
270 	}
271 	
272 	/**
273 	 * Calculate the magnitude of this vector
274 	 * @function
275 	 * @returns {float} The magnitude
276 	 * @since 0.0.0
277 	 */
278 	this.getMagnitude = function() {
279 		return Math.sqrt(this.getMagnitudeSquare());
280 	}
281 	
282 	/**
283 	 * Calculate the squared magnitude of this vector
284 	 * @function
285 	 * @returns {float} The square magnitude
286 	 * @since 0.0.0
287 	 */
288 	this.getMagnitudeSquare = function() {
289 		return this.x * this.x + this.y * this.y;
290 	}
291 	
292 	/**
293 	 * Normalizes this vector
294 	 * @function
295 	 * @returns {Vector2} The normalized form of this vector
296 	 * @since 0.0.0
297 	 */
298 	this.normalize = function() {
299 		var newVector = this.clone();
300 		newVector.normalizeMutate();
301 		return newVector;
302 	}
303 	
304 	/**
305 	 * Normalizes this vector
306 	 * @function
307 	 * @returns {void}
308 	 * @since 0.0.0
309 	 */
310 	this.normalizeMutate = function() {
311 		var magnitude = this.getMagnitude();
312 		if (magnitude > 0) {
313 			this.multScalarMutate(1 / magnitude);
314 		} // if
315 	}
316 	
317 	/**
318 	 * Zero out this vector
319 	 * @function
320 	 * @returns {void}
321 	 * @since 0.0.0
322 	 */
323 	this.zeroMutate = function() {
324 		this.x = 0;
325 		this.y = 0;
326 	}
327 	
328 	/**
329 	 * Clones this vector
330 	 * @function
331 	 * @returns {Vector2} The clone of this vector
332 	 * @since 0.0.0
333 	 */
334 	this.clone = function() {
335 		return new Vector2(this.x, this.y);
336 	}
337 	
338 	/**
339 	 * Converts the class to a string representation
340 	 * @function
341 	 * @returns {string} The string representation of this class
342 	 * @since 0.0.0
343 	 */
344 	this.toString = function() {
345 		return "(" + this.x + ", " + this.y + ")";
346 	}
347 	
348 	/**
349 	 * Draw visual helpers for this vector
350 	 *
351 	 * This method assumes that translations have already been made to the starting
352 	 * position of the vector
353 	 * @function
354 	 * @param {StrokeStyle} strokestyle Optional overriding stroke style
355 	 * @returns {void}
356 	 * @since 0.0.0
357 	 */
358 	this.draw = function(strokestyle) {
359 		ctx.save();	
360 		
361 		ctx.strokeStyle = (strokestyle == undefined ? "lightblue" : strokestyle);
362 		ctx.beginPath();
363 		ctx.moveTo(0, 0);
364 		ctx.lineTo(this.x, this.y);
365 		ctx.stroke();
366 
367 		ctx.restore();
368 	}
369 	
370 	/**
371 	 * Draw visual helpers for this vector
372 	 *
373 	 * This method assumes that translations have already been made to the starting
374 	 * position of the vector
375 	 * @function
376 	 * @param {FillStyle} fillStyle Optional overriding fill style
377 	 * @returns {void}
378 	 * @since 0.0.0
379 	 */
380 	this.drawPoint = function(fillStyle) {
381 		ctx.save();	
382 		ctx.fillStyle = (fillStyle == undefined ? "magenta" : fillStyle);
383 		ctx.translate(this.x, this.y);
384 		ctx.fillRect(
385 			-1, -1, 2, 2
386 		);
387 		ctx.restore();
388 	}
389 }

391 /**
392  * Calculates the angle between two vectors in radians
393  * @function
394  * @static
395  * @param {Vector2} v1 The first vector
396  * @param {Vector2} v2 The second vector
397  * @returns {float} The angle between the two vectors
398  * @since 0.0.0
399  */
400 Vector2.getAngle = function(v1, v2) {
401 	return Math.acos(v1.normalize().dotProduct(v2.normalize()));
402 }

404 /**
405  * @class 2x2 matrix
406  * @constructor
407  * @param {float} data1 Matrix data entry
408  * @param {float} data2 Matrix data entry
409  * @param {float} data3 Matrix data entry
410  * @param {float} data4 Matrix data entry
411  * @since 0.0.0
412  */
413 function Matrix2(data1, data2, data3, data4, inverse) {
414 
415 	/**
416 	 * Matrix entries
417 	 * @field
418 	 * @type float []
419 	 * @default [0.0, 0.0, 0.0, 0.0]
420 	 * @since 0.0.0
421 	 */
422 	this.e = [
423 		data1 || 0.0, data2 || 0.0, data3 || 0.0, data4 || 0.0
424 	];
425 	
426 	/**
427 	 * Matrix determinant
428 	 * @field
429 	 * @type float
430 	 * @default 0.0
431 	 * @since 0.0.0
432 	 */
433 	this.det = 0.0;
434 	
435 	/**
436 	 * Matrix inverse
437 	 * @field
438 	 * @type Matrix2
439 	 * @default undefined
440 	 * @since 0.0.0
441 	 */
442 	this.inverse = undefined;
443 	
444 	/**
445 	 * Calculates this determinant {@link Matrix2#det} from matrix entries {@link Matrix2#e}
446 	 * @function
447 	 * @private
448 	 * @returns {void}
449 	 * @since 0.0.0
450 	 */
451 	this.recaclulateDeterminant = function() {
452 		this.det = this.e[0] * this.e[3] - this.e[1] * this.e[2];
453 	}
454 	this.recaclulateDeterminant();
455 	
456 	/**
457 	 * Calculates this inverse {@link Matrix2#inverse} from matrix entries {@link Matrix2#e}
458 	 * @function
459 	 * @private
460 	 * @returns {void}
461 	 * @since 0.0.0
462 	 */
463 	this.recaclulateInverse = function() {
464 		
465 		if (this.det == 0) {
466 			return;
467 		} // if
468 		
469 		var inverseDet = 1 / this.det;
470 		var i = new Matrix2(
471 			inverseDet * this.e[3],
472 			-inverseDet * this.e[1],
473 			-inverseDet * this.e[2],
474 			inverseDet * this.e[0]
475 		);
476 		i.inverse = this.clone();
477 		this.inverse = i;
478 	}
479 	
480 	/**
481 	 * Multiply this matrix by the supplied 2D vector
482 	 * @function
483 	 * @param {Vector2} v The vector to multiply
484 	 * @returns {Vector2} The resulting vector
485 	 * @since 0.0.0
486 	 */
487 	this.multVector = function(v) {
488 		return new Vector2(
489 			v.x * this.e[0] + v.y * this.e[1],
490 			v.x * this.e[2] + v.y * this.e[3]
491 		);
492 	}
493 	
494 	/**
495 	 * Multiply this matrix by the supplied 2D vector
496 	 * @function
497 	 * @param {Vector2} v The vector to multiply
498 	 * @returns {void}
499 	 * @since 0.0.0
500 	 */
501 	this.multVectorMutate = function(v) {
502 		v.x = v.x * this.e[0] + v.y * this.e[1];
503 		v.y = v.x * this.e[2] + v.y * this.e[3];
504 	}
505 	
506 	/**
507 	 * Multiply this matrix by the supplied 2x2 matrix
508 	 * @function
509 	 * @param {Matrix2} o The matrix to multiply
510 	 * @returns {Matrix2} The resulting matrix
511 	 * @since 0.0.0
512 	 */
513 	this.mult = function(o) {
514 		return new Matrix2(
515 			this.e[0] * o.e[0] + this.e[1] * o.e[2],
516 			this.e[0] * o.e[1] + this.e[1] * o.e[3],
517 			this.e[2] * o.e[0] + this.e[3] * o.e[2],
518 			this.e[2] * o.e[1] + this.e[3] * o.e[3]
519 		);
520 	}
521 	
522 	/**
523 	 * Adds this matrix by the supplied 2x2 matrix
524 	 * @function
525 	 * @param {Matrix2} o The matrix to add
526 	 * @returns {Matrix2} The resulting matrix
527 	 * @since 0.0.0
528 	 */
529 	this.add = function(o) {
530 		return new Matrix2(
531 			this.e[0] + o.e[0],
532 			this.e[1] + o.e[1],
533 			this.e[2] + o.e[2],
534 			this.e[3] + o.e[3]
535 		);
536 	}
537 	
538 	/**
539 	 * Adds this matrix by the supplied 2x2 matrix
540 	 * @function
541 	 * @param {Matrix2} o The matrix to add
542 	 * @returns {void}
543 	 * @since 0.0.0
544 	 */
545 	this.addMutate = function(o) {
546 		this.e[0] += o.e[0];
547 		this.e[1] += o.e[1];
548 		this.e[2] += o.e[2];
549 		this.e[3] += o.e[3];
550 	}
551 	
552 	/**
553 	 * Gets the entry at row i and column j
554 	 * @function
555 	 * @param {int} i The row number, 0 being row 1
556 	 * @param {int} j The column number, 0 being column 1
557 	 * @returns {float} The entry at row i and column j
558 	 * @since 0.0.0
559 	 */
560 	this.getEntry = function(i, j) {
561 		return this.e[i * 2 + j];
562 	}
563 	
564 	/**
565 	 * Gets the determinant of this matrix
566 	 * @function
567 	 * @returns {float} The determinant of this matrix
568 	 * @since 0.0.0
569 	 */
570 	this.getDeterminant = function() {
571 		return this.det;
572 	}
573 	
574 	/**
575 	 * Determines if this matrix is singular (non-invertable, has a zero determinant)
576 	 * @function
577 	 * @returns {boolean} The singular flag of this matrix
578 	 * @since 0.0.0
579 	 * @see Matrix2#isInvertable
580 	 */
581 	this.isSingular = function() {
582 		return this.det == 0;
583 	}
584 	
585 	/**
586 	 * Determines if this matrix is invertable (non-singular, has a non-zero determinant)
587 	 * @function
588 	 * @returns {boolean} The invertable flag of this matrix
589 	 * @since 0.0.0
590 	 * @see Matrix2#isSingular
591 	 */
592 	this.isInvertable = function() {
593 		return this.det != 0;
594 	}
595 	
596 	/**
597 	 * Gets the inverse of this matrix
598 	 * @function
599 	 * @returns {Matrix} The inverse of this matrix, undefined if singular
600 	 * @since 0.0.0
601 	 * @see Matrix2#isSingular
602 	 */
603 	this.getInverse = function() {
604 		if (!this.inverse && this.isInvertable()) {
605 			this.recaclulateInverse();
606 		} // if
607 		
608 		return this.inverse;
609 	}
610 	
611 	/**
612 	 * Determines if this matrix is equal to another matrix
613 	 * For matrices this is true if every element in this
614 	 * matrix has a corresponding and equal entry in the other
615 	 * matrix
616 	 * @function
617 	 * @param {Matrix2} o The matrix to compare
618 	 * @returns {boolean} true if this matrix is equal to the supplied matrix
619 	 * @since 0.0.0
620 	 */
621 	this.equals = function(o) {
622 		if (this === o) {
623 			return true;
624 		} // if
625 		
626 		if (this.e.length != o.e.length) {
627 			return false;
628 		} // if
629 		
630 		for (i in this.e) {
631 			if (this.e[i] != o.e[i]) {
632 				return false;
633 			} // if
634 		} // for
635 		
636 		return true;
637 	}
638 	
639 	/**
640 	 * Clones this matrix
641 	 * @function
642 	 * @return {Matrix2} The cloned version of this matrix
643 	 * @since 0.0.0
644 	 */
645 	this.clone = function() {
646 		return new Matrix2(this.e[0], this.e[1], this.e[2], this.e[3]);
647 	}
648 }

674 /**
675  * @class 2-Dimensional rotation matrix
676  * <p>
677  * The 2D rotation matrix looks as follows:
678  * | cos(theta), -sin(theta) |
679  * | sin(theta), cos(theta)  |
680  * </p>
681  * @constructor
682  * @since 0.0.0
683  */
684 function RotationMatrix2() {
685 
686 	/**
687 	 * Creates the rotation matrix for the supplied angle
688 	 * @function
689 	 * @private
690 	 * @param {float} theta The angle to get the matrix for
691 	 * @returns {Matrix} The rotation matrix for the supplied angle
692 	 * @since 0.0.0
693 	 */
694 	this.getRotationMatrix = function(theta) {
695 		return new Matrix2(
696 			Math.cos(theta),
697 			-Math.sin(theta),
698 			Math.sin(theta),
699 			Math.cos(theta)
700 		);
701 	}
702 	
703 	/**
704 	 * Apply matrix to the supplied vector
705 	 * @function
706 	 * @param {Vector2} vector The vector to rotate
707 	 * @param {float} theta The amount to rotate the vector by
708 	 * @returns {Vector2} The rotated vector
709 	 * @since 0.0.0
710 	 */
711 	this.rotate = function(vector, theta) {
712 		return this.getRotationMatrix(theta).multVector(vector);
713 	}
714 	
715 	/**
716 	 * Apply matrix to the supplied vector
717 	 * @function
718 	 * @param {Vector2} vector The vector to rotate
719 	 * @param {float} theta The amount to rotate the vector by
720 	 * @returns {void}
721 	 * @since 0.0.0
722 	 */
723 	this.rotateMutate = function(vector, theta) {
724 		this.getRotationMatrix(theta).multVectorMutate(vector);
725 	}
726 }

728 /**

732  * @type RotationMatrix2
733  * @default new RotationMatrix2()
734  * @since 0.0.0

737
