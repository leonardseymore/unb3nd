 17 /**
 18  * Calculate window Y-coordinate based on world Y-coordinate
 19  * @function
 20  * @param {float} y World Y-coordinate
 21  * @returns {float} Window Y-coordinate
 22  * @since 0.0.0
 23  */
 24 function Y(y) {
 25 	return windowRect.height - y;
 26 }
 27 
 28 /**
 29  * From window position to world position
 30  * @function
 31  * @param {Vector2} point Window position
 32  * @returns {Vector2} Window position
 33  * @since 0.0.0.3
 34  */
 35 function world(point) {
 36   var worldPos = point.clone();
 37   worldPos.y = Y(point.y);
 38   worldPos.multScalarMutate(1 / ppm);
 39   return worldPos;
 40 }
 41 
 42 /**
 43  * From world position to window position
 44  * @function
 45  * @param {Vector2} point World position
 46  * @returns {Vector2} Window position
 47  * @since 0.0.0.3
 48  */
 49 function window(point) {
 50   var windowPos = point.clone();
 51   windowPos.multScalarMutate(ppm);
 52   windowPos.y = Y(windowPos.y);
 53   return windowPos;
 54 }
 55 
 56 /**
 57  * From world vector to window vector
 58  * @function
 59  * @param {Vector2} vector World vector
 60  * @returns {Vector2} Window vector
 61  * @since 0.0.0.3
 62  */
 63 function windowVector(vector) {
 64   var windowVec = vector.clone();
 65   windowVec.y = -vector.y;
 66   windowVec.multScalarMutate(ppm);
 67   return windowVec;
 68 }
