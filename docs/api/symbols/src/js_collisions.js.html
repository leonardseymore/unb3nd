 15 /**
 16  * @class A bounding sphere 
 17  * @constructor
 18  * @extends BoundingVolume
 19  * @param {Vector2} center Center of the sphere
 20  * @param {float} radius Radius of the sphere
 21  * @since 0.0.0
 22  */
 23 function BoundingSphere(center, radius) {
 24 	
 25 	/** 
 26 	 * Center of the sphere
 27 	 * @field 
 28 	 * @type Vector2
 29 	 * @default undefined
 30 	 * @since 0.0.0
 31 	 */
 32 	this.center = center || undefined;
 33 	
 34 	/** 
 35 	 * Radius of the sphere
 36 	 * @field 
 37 	 * @type float
 38 	 * @default 0.0
 39 	 * @since 0.0.0
 40 	 */
 41 	this.radius = radius || 0.0;
 42 	
 43 	/**
 44 	 * Determines if the bounding sphere overlaps with the supplied bounding
 45 	 * sphere
 46 	 * @function
 47 	 * @param {BoundingSphere} o The other sphere to test
 48 	 * @returns {boolean} True if this overlaps with the supplied bounding 
 49 	 *  sphere
 50 	 * @since 0.0.0
 51 	 */
 52 	this.overlaps = function(o) {
 53 		var distanceSquared = this.center.sub(
 54 			o.center
 55 		).getMagnitudeSquare();
 56 		return distanceSquared < 
 57 			(this.radius + o.radius) * (this.radius + o.radius);
 58 	}
 59 }

124 /**
125  * @class A base class for nodes in a bounding volume hierarchy.  This class
126  *  uses a binary tree to store the bounding volumes.
127  * @constructor
128  * @param {BVHNode []} children The child nodes
129  * @param {BoundingVolume} volume Holds a single volume encompassing all the
130  *  descendents of this node.
131  * @param {RigidBody} rigidBody Holds the rigid body at this node of the 
132  *  hierarchy. Only leaf nodes can have rigid body defined {@link BVHNode#isLeaf}.
133  *  Note that it is possible to rewrite the algorithms in this class to handle 
134  *  objects at all levels of the hierarchy, but the code provided ignores this 
135  *  vector unless firstChild is undefined.
136  * @since 0.0.0
137  */
138 function BVHNode(children, volume, rigidBody) {
139 	
140 	/** 
141 	 * The child nodes
142 	 * @field 
143 	 * @type BVHNode []
144 	 * @default undefined
145 	 * @since 0.0.0
146 	 */
147 	this.children = children || undefined;
148 	
149 	/** 
150 	 * Holds a single volume encompassing all the descendents of this node.
151 	 * @field 
152 	 * @type BoundingVolume
153 	 * @default undefined
154 	 * @since 0.0.0
155 	 */
156 	this.volume = volume || undefined;
157 	
158 	/** 
159 	 * Holds the rigid body at this node of the 
160 	 * hierarchy. Only leaf nodes can have rigid body defined {@link BVHNode#isLeaf}.
161 	 * Note that it is possible to rewrite the algorithms in this class to handle 
162 	 * objects at all levels of the hierarchy, but the code provided ignores this 
163 	 * vector unless firstChild is undefined.
164 	 * @field 
165 	 * @type RigidBody
166 	 * @default undefined
167 	 * @since 0.0.0
168 	 */
169 	this.rigidBody = rigidBody || undefined;
170 	
171 	/**
172 	 * Determines if this node is at the bottom of the hierarchy
173 	 * @function
174 	 * @returns {boolean} True if this node is at the bottom of the hierarchy
175 	 * @since 0.0.0
176 	 */
177 	this.isLeaf = function() {
178 		return (body != undefined);
179 	}
180 	
181 	/**
182 	 * Inserts the given rigid body, with the given bounding volume,
183 	 * into the hierarchy. This may involve the creation of
184 	 * further bounding volume nodes.
185 	 * @function
186 	 * @param {RigidBody} rigidBody The rigid body to add
187 	 * @param {BoundingVolume} volume The type of volume to add
188 	 * @returns {void}
189 	 * @since 0.0.0
190 	 */
191 	this.insert = function(rigidBody, volume) {
192 		// If we are a leaf, then the only option is to spawn two
193 		// new children and place the new body in one.
194 		if (this.isLeaf()) {
195 			this.children[0] = new BVHNode(
196 				this, this.volume, rigidBody
197 			);
198 			
199 			this.children[1] = new BVHNode(
200 				this, volume, rigidBody
201 			);
202 			
203 			this.rigidBody = undefined;
204 			this.recalculateBoundingVolume();
205 		} else {
206 			// Otherwise we need to work out which child gets to keep
207 			// the inserted body. We give it to whoever would grow the
208 			// least to incorporate it.
209 			if (this.children[0].volume.getGrowth(volume) <
210 				this.children[1].volume.getGrowth(volume)) {
211 				this.children[0].insert(rigidBody, volume);
212 			} else {
213 				this.children[1].insert(rigidBody, volume);
214 			} // if
215 		} // if
216 	}
217 	
218 	/**
219  	 * Deletes this node, removing it first from the hierarchy, along
220 	 * with its associated rigid body and child nodes. This method
221 	 * deletes the node and all its children (but obviously not the
222 	 * rigid bodies). This also has the effect of deleting the sibling
223 	 * of this node, and changing the parent node so that it contains
224 	 * the data currently in that sibling. Finally it forces the
225 	 * hierarchy above the current node to reconsider its bounding
226 	 * volume.
227 	 * @function
228 	 * @returns {void}
229 	 * @since 0.0.0
230 	 */
231 	this.remove = function() {
232 		// If we don't have a parent, then we ignore the sibling processing.
233 		if (this.parent) {
234 			var sibling = undefined;
235 			if (this.parent.children[0] === this) {
236 				sibling = this.parent.children[1];
237 			} else {
238 				sibling = this.parent.children[0];
239 			} // if
240 			
241 			this.parent.volume = sibling.volume;
242 			this.parent.rigidBody = sibling.rigidBody;
243 			this.parent.children[0] = sibling.children[0];
244 			this.parent.children[1] = sibling.children[1];
245 			
246 			sibling.parent = undefined;
247 			sibling.rigidBody = undefined;
248 			sibling.children[0] = undefined;
249 			sibling.children[1] = undefined;
250 			sibling.remove();
251 			
252 			this.parent.recalculateBoundingVolume();
253 		} // if
254 		
255 		if (this.children[0]) {
256 			this.children[0].parent = undefined;
257 			this.children[0].remove();
258 		} // if
259 		
260 		if (this.children[1]) {
261 			this.children[1].parent = undefined;
262 			this.children[1].remove();
263 		} // if
264 	}
265 	
266 	/**
267 	 * Checks the potential contacts from this node downward in
268 	 * the hierarchy, writing them to the given array (up to the
269 	 * given limit). Returns the number of potential contacts it
270 	 * found.
271 	 * @function
272 	 * @param {PotentialContact []} potentialContacts The array to append to
273 	 * @param {int} limit The maximum number of contacts that may be generated
274 	 * @returns {int} Number of potential contacts found
275 	 * @since 0.0.0
276 	 */
277 	this.getPotentialContacts = function(potentialContacts, limit) {
278 		if (this.isLeaf() || limit == 0) {
279 			return 0;
280 		} // if
281 		
282 		return this.children[0].getPotentialContactsWith(
283 			this.children[1], potentialContacts, limit
284 		);
285 	}
286 	
287 	/**
288 	 * Gets potential contacts between this node and the supplied node.
289 	 * @function
290 	 * @param {BVHNode} o The other node to use
291 	 * @param {PotentialContact []} potentialContacts The array to append to
292 	 * @param {int} limit The maximum number of contacts that may be generated
293 	 * @returns {int} Number of potential contacts found
294 	 * @since 0.0.0
295 	 */
296 	this.getPotentialContactsWith = function(o, potentialContacts, limit) {
297 		if (!this.overlaps(o) || limit == 0) {
298 			return 0;
299 		} // if
300 		
301 		if (this.isLeaf() && o.isLeaf()) {
302 			var potentialContact = new PotentialContact([
303 				this.rigidBody, o.rigidBody
304 			]);
305 			this.potentialContacts.push(potentialContact);
306 			return 1;
307 		} // if
308 		
309 		// Determine which node to descend into. If either is
310 		// a leaf, then we descend the other. If both are branches,
311 		// then we use the one with the largest size.
312 		if (o.isLeaf() || 
313 		  (!this.isLeaf() && this.volume.getSize() >= o.volume.getSize())) {
314 			// recurse onto self
315 			var count = this.children[0].getPotentialContactsWith(
316 				o, potentialContacts, limit
317 			);
318 			
319 			if (limit > count) {
320 				return count + this.children[1].getPotentialContactsWith(
321 					o, potentialContacts, limit
322 				);
323 			} else {
324 				return count;
325 			} // if
326 		} else {
327 			// recurse onto other
328 			var count = o.children[0].getPotentialContactsWith(
329 				o, potentialContacts, limit
330 			);
331 			
332 			if (limit > count) {
333 				return count + o.children[1].getPotentialContactsWith(
334 					o, potentialContacts, limit
335 				);
336 			} else {
337 				return count;
338 			} // if
339 		} // if
340 	}
341 	
342 	/**
343 	 * Checks if this node overlaps with the supplied node
344 	 * @function
345 	 * @param {BVHNode} o The other node to use
346 	 * @returns {boolean} True if this node overlaps with the supplied node
347 	 * @since 0.0.0
348 	 */
349 	this.overlaps = function(o) {
350 		return this.volume.overlaps(o.volume);
351 	}
352 }
