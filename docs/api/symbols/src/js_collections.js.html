  7 /**
  8  * @class A queue (FIFO) collection implementation
  9  * @constructor
 10  * @since 0.0.0
 11  */
 12 function Queue() {
 13 
 14 	/** 
 15 	 * Array place holder for queue items
 16 	 * @field 
 17 	 * @type Object []
 18 	 * @default []
 19 	 * @since 0.0.0
 20 	 */
 21 	var queue = [];
 22 	
 23 	/**
 24 	 * Gets the size of the queue
 25 	 * @function
 26 	 * @returns {int} The size of the queue
 27 	 * @since 0.0.0
 28 	 */
 29 	this.getSize = function() {
 30 		return queue.length;
 31 	}
 32 	
 33 	/**
 34 	 * Determines if the queue is empty
 35 	 * @function
 36 	 * @returns {boolean} True if the queue is empty
 37 	 * @since 0.0.0
 38 	 */
 39 	this.isEmpty = function() {
 40 		return (queue.length == 0);
 41 	}
 42 	
 43 	/**
 44 	 * Enqueues the specified element in this queue
 45 	 * @function
 46 	 * @param {Object} element The element to push into the queue
 47 	 * @returns {int} The new length of the queue
 48 	 * @since 0.0.0
 49 	 */
 50 	this.enqueue = function(element) {
 51 		return queue.push(element);
 52 	}
 53 	
 54 	/**
 55 	 * Dequeues the oldest element in the queue
 56 	 * @function
 57 	 * @returns {Object} The oldest element in the queue, null if queue is empty
 58 	 * @since 0.0.0
 59 	 */
 60 	this.dequeue = function() {
 61 		return queue.shift();
 62 	}
 63 	
 64 	/**
 65 	 * Gets all elements in this queue
 66 	 * @function
 67 	 * @returns {Object []} All elements in the queue
 68 	 * @since 0.0.0
 69 	 */
 70 	this.getElements = function() {
 71 		return queue;
 72 	}
 73 }

 75 /**
 76  * @class A ringbuffer(FIFO) collection implementation.  The ringbuffer
 77  * will automatically push out old items when maximum size is reached.
 78  * @constructor
 79  * @param {int} size The size of the buffer
 80  * @throws Error if size is not a valid integer
 81  * @since 0.0.0
 82  */
 83 function RingBuffer(size) {
 84 	
 85 	/*
 86 	 * Size must be passed in
 87 	 */
 88 	if (typeof(size) == "undefined") {
 89 		throw new Error("Valid integer size expected");
 90 	} // if
 91 
 92 	/** 
 93 	 * Array place holder for queue items
 94 	 * @field 
 95 	 * @type Queue
 96 	 * @default new Queue()
 97 	 * @since 0.0.0
 98 	 */
 99 	var queue = new Queue();
100 	
101 	/**
102 	 * Gets the size of the queue
103 	 * @function
104 	 * @returns {int} The size of the queue
105 	 * @since 0.0.0
106 	 */
107 	this.getSize = function() {
108 		return queue.getSize();
109 	}
110 	
111 	/**
112 	 * Determines if the queue is empty
113 	 * @function
114 	 * @returns {boolean} True if the queue is empty
115 	 * @since 0.0.0
116 	 */
117 	this.isEmpty = function() {
118 		return queue.isEmpty();
119 	}
120 	
121 	/**
122 	 * Enqueues the specified element in this queue
123 	 * @function
124 	 * @param {Object} element The element to push into the queue
125 	 * @returns {int} The new length of the queue
126 	 * @since 0.0.0
127 	 */
128 	this.enqueue = function(element) {
129 		if (this.getSize() >= size) {
130 			queue.dequeue();
131 		} // if
132 		
133 		return queue.enqueue(element);
134 	}
135 	
136 	/**
137 	 * Dequeues the oldest element in the queue
138 	 * @function
139 	 * @returns {Object} The oldest element in the queue, null if queue is empty
140 	 * @since 0.0.0
141 	 */
142 	this.dequeue = function() {
143 		return queue.dequeue();
144 	}
145 	
146 	/**
147 	 * Gets all elements in this buffer
148 	 * @function
149 	 * @returns {Object []} All elements in the buffer
150 	 * @since 0.0.0
151 	 */
152 	this.getElements = function() {
153 		return queue.getElements();
154 	}
155 }

157 /**
158  * @class A set of unique elements
159  * @constructor
160  * @since 0.0.0
161  */
162 function Set() {
163 
164 	/**
165 	 * All elements in the set
166 	 * @field
167 	 * @type Object []
168 	 * @default []
169 	 * @since 0.0.0
170 	 */
171 	this.elements = [];
172 	
173 	/**
174 	 * Add a new element to the set
175 	 * @function
176 	 * @param {Object} element The element to add
177 	 * @returns {int} Number of elements
178 	 * @since 0.0.0
179 	 */
180 	this.add = function(element) {
181 		return this.elements.push(element);
182 	}
183 	
184 	/**
185 	 * Remove an element at the specified index
186 	 * @function
187 	 * @param {int} index The index of the element
188 	 * @returns {Object} The removed element
189 	 * @since 0.0.0
190 	 */
191 	this.removeEntityAtIndex = function(index) {
192 		return this.elements.splice(index, 1);
193 	}
194 	
195 	/**
196 	 * Remove the specified element from the set
197 	 * @function
198 	 * @param {Object} element The element to remove
199 	 * @returns {boolean} True if the element was removed
200 	 * @since 0.0.0
201 	 */
202 	this.remove = function(element) {
203 		for (i in this.elements) {
204 			var el = this.elements[i];
205 			if (el === element) {
206 				this.removeElementAtIndex(i);
207 				return true;
208 			} // if
209 		} // for
210 		return false;
211 	}
212 }
