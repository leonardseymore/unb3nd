  7 /**
  8  * @class A rigid body consists of a number of points in space. In this engine
  9  *        calculations on rigid bodies are done on the centre of mass. This 
 10  *        greatly simplifies the mathematics involved in calculating the 
 11  *        relative velocities of each point in space occupied by the rigid body.
 12  * @constructor
 13  * @extends Observable
 14  * @param {float} mass Optional mass, 0.0 being infinite
 15  * @param {float} inertia Optional inertia, 0.0 being infinite
 16  * @since 0.0.0
 17  */
 18 function RigidBody(mass, inertia) {
 19 
 20 	/*
 21 	 * Super constructor
 22 	 */
 23 	Observable.call(this);
 24 	
 25 	/**
 26 	 * Unique identifier to help track a single rigid body
 27 	 * @field 
 28 	 * @type long
 29 	 * @default Auto number
 30 	 * @since 0.0.0
 31 	 * @see RigidBody#getNextUid 
 32 	 */
 33 	this.uid = RigidBody.getNextUid();
 34 	
 35 	/**
 36 	 * The orientation, should only be set using the 
 37 	 * {@link RigidBody#setOrientation} method
 38 	 * @field 
 39 	 * @private
 40 	 * @type Vector2
 41 	 * @default X-unit vector
 42 	 * @since 0.0.0
 43 	 */
 44 	this.orientation = new Vector2(1, 0);
 45 
 46 	/**
 47 	 * The angular velocity (anti-clockwise) of the rigid body around the 
 48 	 * center of gravity
 49 	 * @field 
 50 	 * @type float
 51 	 * @default 0.0
 52 	 * @since 0.0.0
 53 	 */
 54 	this.angVel = 0.0;
 55 	
 56 	/**
 57 	 * The position of this rigid body's center of mass
 58 	 * @field 
 59 	 * @type Vector2
 60 	 * @default Zero vector
 61 	 * @since 0.0.0
 62 	 */
 63 	this.pos = new Vector2();
 64 	
 65 	/**
 66 	 * The velocity of this rigid body's center of mass
 67 	 * @field
 68 	 * @type Vector2 
 69 	 * @default Zero vector
 70 	 * @since 0.0.0
 71 	 */
 72 	this.vel = new Vector2();
 73 	
 74 	/**
 75 	 * The acceleration of this rigid body's center of mass
 76 	 * @field
 77 	 * @type Vector2 
 78 	 * @default Zero vector
 79 	 * @since 0.0.0
 80 	 */
 81 	this.acc = new Vector2();
 82 	
 83 	/**
 84 	 * Damping is a simple yet special property involved in slowing down moving
 85 	 * objects 0 - 1, where 0 is full damping and 1 is no damping
 86 	 * @field
 87 	 * @type float 
 88 	 * @default 1.0
 89 	 * @since 0.0.0
 90 	 */
 91 	this.damping = 1.0;
 92 	
 93 	/**
 94 	 * Angular damping does for rotations what {@link #damping} does for linear
 95 	 * motion
 96 	 * @field
 97 	 * @type float 
 98 	 * @default 1.0
 99 	 * @since 0.0.0
100 	 */
101 	this.angularDamping = 1.0;
102 	
103 	/**
104 	 * Holds the inverse mass
105 	 * @private
106 	 * @field
107 	 * @type float 
108 	 * @default 0.0, indicating infinite mass
109 	 * @since 0.0.0
110 	 */
111 	this.inverseMass = 0.0;
112 	if (mass) {
113 		this.inverseMass = 1 / mass;
114 	} // if
115 	
116 	/**
117 	 * Holds the inverse inertia over the center of gravity along the Z-axis
118 	 * @private
119 	 * @field
120 	 * @type float 
121 	 * @default 0.0, indicating infinite inertia
122 	 * @since 0.0.0
123 	 */
124 	this.inverseInertia = 0.0;
125 	if (inertia) {
126 		this.inverseInertia = 1 / inertia;
127 	}
128 	
129 	/**
130 	 * The overall force accumulator
131 	 * @private
132 	 * @field
133 	 * @type Vector2 
134 	 * @default Zero vector
135 	 * @since 0.0.0
136 	 */
137 	this.forceAccum = new Vector2();
138 	
139 	/**
140 	 * The overall torque accumulator
141 	 * @private
142 	 * @field
143 	 * @type float 
144 	 * @default 0.0
145 	 * @since 0.0.0
146 	 */
147 	this.torqueAccum = 0.0;
148 	
149 	/**
150 	 * Holds a transform matrix for converting body space into world
151 	 * space and vice versa. This can be achieved by calling the
152 	 * getPointIn*Space functions.
153 	 *
154 	 * Data cache of position and orientation calculated once per
155 	 * frame.
156 	 *
157 	 * @field 
158 	 * @type Matrix2
159 	 * @default Zero matrix
160 	 * @since 0.0.0
161 	 */
162 	this.transformMatrix = undefined;
163 	
164 	/**
165 	 * Calculates internal data from state data. This should be called
166 	 * after the body's state is altered directly (it is called
167 	 * automatically during integration). If you change the body's
168 	 * state and then intend to integrate before querying any data
169 	 * (such as the transform matrix), then you can omit this step.
170 	 * @function
171 	 * @returns {void}
172 	 * @since 0.0.0
173 	 */
174 	this.calculateDerivedData = function() {
175 		this.calculateTransformMatrix();
176 	}
177 	
178 	/**
179 	 * Calculates the {@link #transformMatrix} from this position and orientation
180 	 * @function
181 	 * @returns {void}
182 	 * @since 0.0.0
183 	 */
184 	this.calculateTransformMatrix = function() {
185 		this.transformMatrix = new TransformationMatrix3(
186       this.getOrientationAngle(), this.pos.x, this.pos.y
187     );
188 	}
189 	
190 	/**
191 	 * Sets the orientation angle of this object (-PI;PI]
192 	 * TODO: this method normalizes the vector after every call
193 	 *       we need a less expensive way to periodically normalize
194 	 *       the orientation vector
195 	 * @function
196 	 * @param {float} theta The orientation to use
197 	 * @returns {void}
198 	 * @since 0.0.0
199 	 */
200 	this.setOrientationAngle = function(theta) {
201 		this.orientation.x = Math.cos(theta);
202 		this.orientation.y = Math.sin(theta);
203 		this.orientation.normalizeMutate();
204 	}
205 	
206 	/**
207 	 * Sets the orientation of this object
208 	 * @function
209 	 * @param {Vector2d} orientation The orientation of this body
210 	 * @returns {void}
211 	 * @since 0.0.0
212 	 */
213 	this.setOrientation = function(orientation) {
214 		this.orientation = orientation;
215 	}
216 	
217 	/**
218 	 * Sets the orientation angle of this object
219 	 * @function
220 	 * @returns {float} The orientation of this object in radians
221 	 * @since 0.0.0
222 	 */
223 	this.getOrientationAngle = function() {
224     var x = this.orientation.x;
225     var y = this.orientation.y;
226 
227     return Math.atan2(y, x);
228 	}
229 	
230 	/**
231 	 * Gets the orientation of this object
232 	 * @function
233 	 * @returns {Vector2} The orientation of this object
234 	 * @since 0.0.0
235 	 */
236 	this.getOrientation = function() {
237 		return this.orientation;
238 	}
239 	
240 	/**
241 	 * Converts a point relative to this center of gravity to world coordinates
242 	 * @function 
243 	 * @param {Vector2} point Point to transform
244 	 * @returns {Vector2} The transformed point
245 	 * @since 0.0.0
246 	 */
247 	 this.getPointInWorldSpace = function(point) {
248      var v3 = new Vector3(point.x, point.y, 1);
249 		 v3 = this.transformMatrix.multVector(v3);
250      return new Vector2(v3.x, v3.y);
251 	 }
252 	 
253 	/**
254 	 * Converts a point in world coordinates to a point relative to this body's local 
255 	 * coordinates
256 	 * @function 
257 	 * @param {Vector2} point Point to transform
258 	 * @returns {Vector2} The transformed point
259 	 * @since 0.0.0
260 	 */
261 	 this.getPointInLocalSpace = function(point) {
262      var v3 = new Vector3(point.x, point.y, 1);
263 		 v3 = this.transformMatrix.getInverse().multVector(v3);
264      return new Vector2(v3.x, v3.y);
265 	 }
266 	
267 	/**
268 	 * Ensures this body has a finite mass
269 	 * @function 
270 	 * @returns {boolean} true if this body has a finite mass
271 	 * @since 0.0.0
272 	 */
273 	 this.hasFiniteMass = function() {
274 		return this.inverseMass > 0;
275 	 }
276 	
277 	/**
278 	 * Adds an external force to the center of mass of this rigid body
279 	 * @function 
280 	 * @param {Vector2} force The force to add to this rigid body
281 	 * @returns {void}
282 	 * @since 0.0.0
283 	 */
284 	this.applyForce = function(force) {
285 		this.forceAccum.addMutate(force);
286 	}
287 	
288 	/**
289 	 * Adds the given force to the given point on the rigid body.
290 	 * The direction of the force is given in world coordinates,
291 	 * but the application point is given in body space. This is
292 	 * useful for spring forces, or other forces fixed to the
293 	 * body.
294 	 *
295 	 * @function 
296 	 * @param {Vector2} force The force to add to this rigid body
297 	 * @param {Vector2} point The location at which to apply the force in world coordinates
298 	 * @returns {void}
299 	 * @since 0.0.0
300 	 */
301 	this.applyForceAtBodyPoint = function(force, point) {
302 		var pt = this.getPointInWorldSpace(point);
303 		this.addForceAtPoint(force, pt);
304 	}
305 	
306 	/**
307 	 * Adds the given force to this body at the relative point
308 	 * @function 
309 	 * @param {Vector2} force The force to add to this rigid body
310 	 * @param {Vector2} point The location at which to apply the force in local coordinates
311 	 * @returns {void}
312 	 * @since 0.0.0
313 	 */
314 	this.addForceAtPoint = function(force, point) {
315 		this.applyForce(force);
316 
317     var pt = point.clone();
318     pt.subMutate(this.pos);
319 		this.applyTorque(
320       pt.vectorProduct(force)
321 		);
322 	}
323 	
324 	/**
325 	 * Reset the force accumulator on this rigid body
326 	 * @function 
327 	 * @returns {void}
328 	 * @since 0.0.0
329 	 */
330 	this.clearForceAccum = function() {
331 		this.forceAccum.zeroMutate();
332 	}
333 	
334 	/**
335 	 * Adds an external torque to the rigid body
336 	 * @function 
337 	 * @param {float} torque The torque to add to this rigid body
338 	 * @returns {void}
339 	 * @since 0.0.0
340 	 */
341 	this.applyTorque = function(torque) {
342 		this.torqueAccum += torque;
343 	}
344 	
345 	/**
346 	 * Reset the torque accumulator on this rigid body
347 	 * @function 
348 	 * @returns {void}
349 	 * @since 0.0.0
350 	 */
351 	this.clearTorqueAccum = function() {
352 		this.torqueAccum = 0.0;
353 	}
354 	
355 	/**
356 	 * Clears all frame accumulators
357 	 * @function 
358 	 * @returns {void}
359 	 * @since 0.0.0
360 	 */
361 	this.clearAccums = function() {
362 		this.clearForceAccum();
363 		this.clearTorqueAccum();
364 	}
365 	
366 	/**
367 	 * Sets the inverse mass of the rigid body
368 	 * @function 
369 	 * @param {float} inverseMass The inverse mass of the rigid body
370 	 * @returns {void}
371 	 * @since 0.0.0
372 	 */
373 	this.setInverseMass = function(inverseMass) {
374 		this.inverseMass = inverseMass;
375 	}
376 	
377 	/**
378 	 * Sets the mass of the rigid body
379 	 * @function
380 	 * @param {float} mass The mass of the rigid body
381 	 * @returns {void}
382 	 * @since 0.0.0
383 	 */
384 	this.setMass = function(mass) {
385 		if (mass == 0) {
386 			this.inverseMass = 0;
387 		} else {
388 			this.inverseMass = 1 / mass;
389 		} // if
390 	}
391 	
392 	/**
393 	 * Gets the mass of the rigid body
394 	 * @function
395 	 * @returns {float} The mass of the rigid body
396 	 * @since 0.0.0
397 	 */
398 	this.getMass = function() {
399 		if (this.inverseMass == 0) {
400 			return 0;
401 		} else {
402 			return 1 / this.inverseMass;
403 		} // if
404 	}
405 	
406 	/**
407 	 * Sets the inverse inertia of the rigid body
408 	 * @function 
409 	 * @param {float} inverseInertia The inverse inertia of the rigid body
410 	 * @returns {void}
411 	 * @since 0.0.0
412 	 */
413 	this.setInverseInertia = function(inverseInertia) {
414 		this.inverseInertia = inverseInertia;
415 	}
416 	
417 	/**
418 	 * Sets the inertia of the rigid body
419 	 * @function
420 	 * @param {float} inertia The inertia of the rigid body
421 	 * @returns {void}
422 	 * @since 0.0.0
423 	 */
424 	this.setInertia = function(inertia) {
425 		if (inertia == 0) {
426 			this.inverseInertia = 0;
427 		} else {
428 			this.inverseInertia = 1 / inertia;
429 		} // if
430 	}
431 	
432 	/**
433 	 * Gets the inertia of the rigid body
434 	 * @function
435 	 * @returns {float} The inertia of the rigid body
436 	 * @since 0.0.0
437 	 */
438 	this.getInertia = function() {
439 		if (this.inverseInertia == 0) {
440 			return 0;
441 		} else {
442 			return 1 / this.inverseInertia;
443 		} // if
444 	}
445 	
446 	/**
447 	 * Integrates the rigid body forward in time by the given amount.
448 	 * Uses Newton-Euler integration function
449 	 * @function
450 	 * @param {int} delta The time delta in milliseconds
451 	 * @returns {void}
452 	 * @since 0.0.0
453 	 */
454 	this.integrate = function(delta) {		
455 		var dt = delta / 1000;
456 		
457 		// linear motion
458 		this.pos.addMutate(
459 			this.vel.multScalar(dt)
460 		);
461 
462 		var resultingAcc = this.acc.add(
463 			this.forceAccum.multScalar(this.inverseMass)
464 		);
465 		
466 		this.vel.addMutate(
467 			resultingAcc.multScalar(dt)
468 		);
469 		
470 		this.vel.multScalarMutate(
471 			Math.pow(this.damping, dt)
472 		);
473 		
474 		// angular motion
475 		var angle = this.getOrientationAngle();
476 		this.setOrientationAngle(
477 			angle + this.angVel * dt
478 		);
479 		
480 		var angularAcc = this.torqueAccum * this.inverseInertia * dt;
481 		this.angVel += angularAcc;
482 		this.angVel *= Math.pow(this.angularDamping, dt);
483 		
484 		this.calculateDerivedData();
485 		this.clearAccums();
486 	}

497   }

503 	 * @since 0.0.0

505 	this.ondie = undefined;
506 
507 	/**

511 	 * @since 0.0.0
512 	 */
513 	this.die = function() {		
514 		if (this.ondie) {
515 			this.ondie();
516 		} // if
517 		this.dispatchEvent("die");
518 	}
519 	
520 	/**
521 	 * Converts the class to a string representation
522 	 * @function
523 	 * @returns {string} The string representation of this class
524 	 * @since 0.0.0
525 	 */
526 	this.toString = function() {
527 		return "uid=" + this.uid;
528 	}

597 /**
598  * @class A gravitational force generator
599  * @constructor
600  * @extends ForceGenerator
601  * @param {Vector2} gravition Gravitational force, defaults {@link DEFAULT_GRAVITATIONAL_CONSTANT}
602  * @since 0.0.0
603  */
604 function GravityForceGenerator(gravitation) {
605 
606 	/**
607 	 * The gravitational pull
608 	 * @field 
609 	 * @type Vector2
610 	 * @default Vector with {@link DEFAULT_GRAVITATIONAL_CONSTANT} as Y-axis
611 	 * @since 0.0.0
612 	 */
613 	this.gravitation = gravitation || new Vector2(0, DEFAULT_GRAVITATIONAL_CONSTANT);
614 	
615 	/**
616 	 * Apply gravity to the given mass over the delta time
617 	 * @function
618 	 * @override
619 	 * @param {RigidBody} rigidBody The rigid body to apply the force to
620 	 * @param {int} delta The delta time in milliseconds
621 	 * @returns {void}
622 	 * @since 0.0.0
623 	 */
624 	this.applyForce = function(rigidBody, delta) {
625 		if (!rigidBody.hasFiniteMass()) {
626 			if (debug && verbose) {
627 				console.debug("RigidBody %o has zero mass", rigidBody);
628 			} // if
629 			return;
630 		} // if
631 		
632 		rigidBody.applyForce(
633 			// scale force, since gravity is constant regardless of mass 
634 			this.gravitation.multScalar(rigidBody.getMass())
635 		);
636 	}
637 
638   /**
639 	 * Accepts the supplied rigid boy visitor
640 	 * @function
641 	 * @override
642 	 */
643 	this.accept = function(visitor, rigidBody) {
644     visitor.visitGravityForceGenerator(this, rigidBody);
645   }
646 	
647 	/**
648 	 * Converts the class to a string representation
649 	 * @function
650 	 * @returns {string} The string representation of this object
651 	 * @since 0.0.0
652 	 */
653 	this.toString = function() {
654 		return "Gravity: " + this.gravitation.toString();
655 	}
656 }

659 /**
660  * @class A spring force generator
661  * @constructor
662  * @extends ForceGenerator
663  * @param {Vector2} connectionPoint The connection point of the spring, in
664  *        local coordinates of this rigid body
665  * @param {RigidBody} rigidBodyOther Rigid body at the other end of the spring
666  * @param {Vector2} connectionPointOther The connection point of the spring,
667  *        in that object's local coordinates of the {@link #rigidBodyOther}
668  * @param {float} springConstant Holds the spring constant
669  * @param {float} restLength Holds the spring's rest length
670  * @since 0.0.0
671  */
672 function SpringForceGenerator(connectionPoint, rigidBodyOther, connectionPointOther,
673 	springConstant, restLength) {
674 	
675 	/**
676 	 * The connection point of the spring, in local coordinates of this rigid 
677 	 * body
678 	 * @field 
679 	 * @type Vector2d
680 	 * @default connectionPoint
681 	 * @since 0.0.0
682 	 */
683 	this.connectionPoint = connectionPoint;
684 	
685 	/**
686 	 * The other connection point of the spring, in local coordinates of the
687 	 * {@link #rigidBodyOther}
688 	 * @field 
689 	 * @type Vector2d
690 	 * @default connectionPointOther
691 	 * @since 0.0.0
692 	 */
693 	this.connectionPointOther = connectionPointOther;
694 
695 	/**
696 	 * Rigid body at the other end of the spring
697 	 * @field
698 	 * @type RigidBody
699 	 * @default rigidBodyOther
700 	 * @since 0.0.0
701 	 */
702 	this.rigidBodyOther = rigidBodyOther;
703 
704 	/**
705 	 * Holds the spring constant
706 	 * @field
707 	 * @type float
708 	 * @default springConstant
709 	 * @since 0.0.0
710 	 */
711 	this.springConstant = springConstant;
712 	
713 	/**
714 	 * Holds the spring's rest length
715 	 * @field
716 	 * @type float
717 	 * @default restLength
718 	 * @since 0.0.0
719 	 */
720 	this.restLength = restLength;
721 	
722 	/**
723 	 * Apply spring force to the given rigid body over the delta time
724 	 * @function
725 	 * @override
726 	 * @param {RigidBody} rigidBody The rigid body to apply the force to
727 	 * @param {int} delta The delta time in milliseconds
728 	 * @returns {void}
729 	 * @since 0.0.0
730 	 */
731 	this.applyForce = function(rigidBody, delta) {
732 		var lws = rigidBody.getPointInWorldSpace(this.connectionPoint);
733 		var ows = this.rigidBodyOther.getPointInWorldSpace(this.connectionPointOther);
734 		var force = lws.sub(ows);
735 		
736 		var magnitude = force.getMagnitude();
737 		magnitude = Math.abs(magnitude - this.restLength);
738 		magnitude *= this.springConstant;
739 		
740 		force.normalizeMutate();
741 		force.multScalarMutate(-magnitude);
742 		rigidBody.addForceAtPoint(force, lws);
743 	}
744 
745   /**
746 	 * Accepts the supplied rigid boy visitor
747 	 * @function
748 	 * @override
749 	 */
750 	this.accept = function(visitor, rigidBody) {
751     visitor.visitSpringForceGenerator(this, rigidBody);
752   }
753 }

1185  * contact removes their interpenetration, and applies sufficient
1186  * impulse to keep them apart. Colliding bodies may also rebound.
1187  *
1188  * Colliding bodies may also rebound. Contacts can be used to represent

1190  * in their correct orientation.
1191  *
1192  * @constructor
1193  * @since 0.0.0.4
1194  */
1195 function Contact() {
1196 
1197 	/**
1198 	 * Holds the position of the contact in world coordinates
1199 	 * @field
1200 	 * @type Vector2
1201 	 * @default undefined
1202 	 * @since 0.0.0.4
1203 	 */
1204 	this.contactPoint = undefined;
1205 
1206 	/**
1207 	 * Holds the direction of the contact in world coordinates
1208 	 * @field
1209 	 * @type Vector2
1210 	 * @default undefined
1211 	 * @since 0.0.0.4
1212 	 */
1213 	this.contactNormal = undefined;
1214 
1215 	/**
1216 	 * Penetration depth in the direction of the contact normal
1217 	 * @field
1218 	 * @type float
1219 	 * @default 0.0
1220 	 * @since 0.0.0.4
1221 	 */
1222 	this.penetration = 0.0;
1223 
1224 	/**
1225 	 * Resolves this contact for both veolcity and interpenetration
1226 	 * @function
1227 	 * @protected
1228 	 * @param {int} delta Delta time in milliseconds since last update
1229 	 * @returns {void}
1230 	 * @since 0.0.0
1231 	 */
1232 	this.resolve = function(delta) {
1233 		this.resolveVelocity(delta);
1234 		this.resolveInterpenetration(delta);
1235 	}
1236 
1237 	/**
1238 	 * Calculate separating velocity at this contact
1239 	 * @function
1240 	 * @protected
1241 	 * @returns {float} Separating velocity at this contact
1242 	 * @since 0.0.0
1243 	 */
1244 	this.calculateSeparatingVelocity = function() {
1245 		var relativeVelocity = this.particles[0].vel.clone();
1246 		if (this.particles[1]) {
1247 			relativeVelocity.subMutate(
1248 				this.particles[1].vel
1249 			);
1250 		} // if
1251 		return relativeVelocity.dotProduct(this.contactNormal);
1252 	}
1253 
1254 	/**
1255 	 * Handles impulse calculations for this collision
1256 	 * @function
1257 	 * @private
1258 	 * @param {int} delta Delta time in milliseconds since last update
1259 	 * @returns {void}
1260 	 * @since 0.0.0
1261 	 */
1262 	this.resolveVelocity = function(delta) {
1263 		var dt = delta / 1000;
1264 		var separatingVelocity = this.calculateSeparatingVelocity();
1265 		if (separatingVelocity > 0) {
1266 			// contact is either separating or stationary
1267 			return;
1268 		} // if
1269 		var newSepVelocity = -separatingVelocity * this.restitution;
1270 
1271 		// ~PHYLIMIT keep to objects in resting contact
1272 		// micro-collision approach
1273 		// check velocity build up due to acceleration only
1274 		var accCausedVelocity = this.particles[0].acc.clone();
1275 		if (this.particles[1]) {
1276 			accCausedVelocity.subMutate(this.particles[1].acc);
1277 		} // if
1278 		var accCausedSepVelocity = accCausedVelocity.dotProduct(this.contactNormal) * dt;
1279 
1280 		// If we've got a closing velocity due to acceleration build-up,
1281 		// remove it from the new separating velocity.
1282 		if (accCausedSepVelocity < 0) {
1283 			newSepVelocity += resitution * accCausedSepVelocity;
1284 
1285 			// Make sure we haven't removed more than was there to remove.
1286 			if (newSepVelocity < 0) {
1287 				newSepVelocity = 0;
1288 			} // if
1289 		} // if
1290 		// ~END-PHYLIMIT
1291 
1292 		var deltaVelocity = newSepVelocity - separatingVelocity;
1293 
1294 		var totalInverseMass = this.particles[0].inverseMass;
1295 		if (this.particles[1]) {
1296 			totalInverseMass += this.particles[1].inverseMass;
1297 		} // if
1298 
1299 		if (totalInverseMass <= 0) {
1300 			// all particles have infinite mass
1301 			return;
1302 		} // if
1303 
1304 		var impulse = deltaVelocity / totalInverseMass;
1305 		var impulsePerIMass = this.contactNormal.multScalar(impulse);
1306 
1307 		// apply impulses proportional to inverse mass
1308 		// in the direction of the contact
1309 		this.particles[0].vel.addMutate(
1310 			impulsePerIMass.multScalar(this.particles[0].inverseMass)
1311 		);
1312 
1313 		if (this.particles[1]) {
1314 			this.particles[1].vel.addMutate(
1315 				impulsePerIMass.multScalar(-this.particles[1].inverseMass)
1316 			);
1317 		} // if
1318 	}
1319 
1320 	/**
1321 	 * Handles interpenetration resolution for this contact
1322 	 * @function
1323 	 * @param {int} delta Delta time in milliseconds since last update
1324 	 * @returns {void}
1325 	 * @since 0.0.0
1326 	 */
1327 	this.resolveInterpenetration = function(delta) {
1328 		if (this.penetration <= 0) {
1329 			return;
1330 		} // if
1331 
1332 		var totalInverseMass = this.particles[0].inverseMass;
1333 		if (this.particles[1]) {
1334 			totalInverseMass += this.particles[1].inverseMass;
1335 		} // if
1336 
1337 		if (totalInverseMass <= 0) {
1338 			// infinite mass
1339 			return;
1340 		} // if
1341 
1342 		// find penetration resolution relative to inverse mass
1343 		var movePerIMass = this.contactNormal.multScalar(
1344 			-this.penetration / totalInverseMass
1345 		);
1346 
1347 		this.particles[0].pos.addMutate(
1348 			movePerIMass.multScalar(this.particles[0].inverseMass)
1349 		);
1350 
1351 		if (this.particles[1]) {
1352 			this.particles[1].pos.addMutate(
1353 				movePerIMass.multScalar(this.particles[1].inverseMass)
1354 			);
1355 		} // if
1356 	}

1576 				this.children[1].volume.getGrowth(volume)) {
1577 				this.children[0].insert(rigidBody, volume);
1578 			} else {

1589 	 * of this node, and changing the parent node so that it contains

1591 	 * hierarchy above the current node to reconsider its bounding
1592 	 * volume.
1593 	 * @function

1610 			this.parent.children[1] = sibling.children[1];

1658 	 * @param {int} limit The maximum number of contacts that may be generated
1659 	 * @returns {int} Number of potential contacts found
1660 	 * @since 0.0.0
1661 	 */
1662 	this.getPotentialContactsWith = function(o, potentialContacts, limit) {
1663 		if (!this.overlaps(o) || limit == 0) {
1664 			return 0;
1665 		} // if
1666 
1667 		if (this.isLeaf() && o.isLeaf()) {

1669 				this.rigidBody, o.rigidBody

1673 		} // if

1676 		// a leaf, then we descend the other. If both are branches,
1677 		// then we use the one with the largest size.
1678 		if (o.isLeaf() ||
1679 		  (!this.isLeaf() && this.volume.getSize() >= o.volume.getSize())) {

1682 				o, potentialContacts, limit
1683 			);
1684

1686 				return count + this.children[1].getPotentialContactsWith(
1687 					o, potentialContacts, limit
1688 				);
1689 			} else {
1690 				return count;
1691 			} // if
1692 		} else {
1693 			// recurse onto other
1694 			var count = o.children[0].getPotentialContactsWith(
1695 				o, potentialContacts, limit
1696 			);
1697 
1698 			if (limit > count) {
1699 				return count + o.children[1].getPotentialContactsWith(
1700 					o, potentialContacts, limit
1701 				);
1702 			} else {
1703 				return count;
1704 			} // if
1705 		} // if
1706 	}
