571 /**
572  * unb3nd 2x2 matrix namespace
573  * @since 0.0.0.4
574  */
575 unb3nd.m2 = {
576 
577   /**
578    * Creates a new 2x2 matrix
579    * @function
580    * @param {Array} src Optional to copy values from
581    * @returns {Array} New matrix
582    * @since 0.0.0.4
583    */
584   create : function(src) {
585     if (src) {
586       return [src[0], src[1], src[2], src[3]];
587     } else {
588       return [0, 0, 0, 0];
589     } // if
590   },
591 
592   /**
593    * Creates a new 2x2 identity matrix
594    * @function
595    * @returns {Array} New identity matrix
596    * @since 0.0.0.4
597    */
598   createIdentity : function() {
599     return [1, 0, 0, 1];
600   },
601 
602   /**
603    * Creates a new 2x2 rotation matrix
604    * <p>
605    * The 2D rotation matrix looks as follows:
606    * | cos(theta), -sin(theta) |
607    * | sin(theta), cos(theta)  |
608    * </p>
609    * @function
610    * @param {Number} theta The angle of this rotation matrix
611    * @returns {Array} New rotation matrix
612    * @since 0.0.0.4
613    */
614   createRotation : function(theta) {
615     return [Math.cos(theta), -Math.sin(theta),
616       Math.sin(theta), Math.cos(theta)];
617   },
618 
619   /**
620    * Calculates the determinant
621    * @function
622    * @param {Array} m The matrix to get the determinant for
623    * @returns {Number} The determinant of the supplied matrix
624    * @since 0.0.0.4
625    */
626   getDeterminant : function(m) {
627     return m[0] * m[3] - m[1] * m[2];
628   },
629 
630   /**
631    * Determines if the supplied matrix is invertable
632    * @function
633    * @param {Array} m The matrix to use
634    * @returns {Boolean} True if the matrix is invertable
635    * @since 0.0.0.4
636    */
637   isInvertable : function(m) {
638     var determinant = this.getDeterminant(m);
639     return determinant != 0;
640   },
641 
642   /**
643    * Determines if the supplied matrix is singular
644    * @function
645    * @param {Array} m The matrix to use
646    * @returns {Boolean} True if the matrix is singular
647    * @since 0.0.0.4
648    */
649   isSingular : function(m) {
650     var determinant = this.getDeterminant(m);
651     return determinant == 0;
652   },
653 
654   /**
655    * Calculates the inverse of the supplied matrix
656    * @function
657    * @param {Array} m The matrix to get the determinant for
658    * @returns {Array} The inverse of the matrix, or undefined if inverse does not exist
659    * @since 0.0.0.4
660    */
661   getInverse : function(m) {
662     var determinant = this.getDeterminant(m);
663     if (determinant == 0) {
664       return undefined;
665     } // if
666 
667     var inverseDet = 1 / determinant;
668     return [inverseDet * m[3],
669       -inverseDet * m[1],
670       -inverseDet * m[2],
671       inverseDet * m[0]];
672   },
673 
674   /**
675    * Multiply the matrix by the supplied vector
676    * @function
677    * @param {Array} m The matrix to use
678    * @param {Array} v The vector to multiply
679    * @returns {Array} The resulting vector
680    * @since 0.0.0.4
681    */
682   multVector : function(m, v) {
683     return [v[0] * m[0] + v[1] * m[1],
684       v[0] * m[2] + v[1] * m[3]];
685   },
686 
687   /**
688    * Multiply the matrix by the supplied vector and mutate the vector
689    * @function
690    * @param {Array} m The matrix to use
691    * @param {Array} v The vector to multiply
692    * @returns {void}
693    * @since 0.0.0.4
694    */
695   multVectorMutate : function(m, v) {
696     v[0] = v[0] * m[0] + v[1] * m[1];
697     v[1] = v[0] * m[2] + v[1] * m[3];
698   },
699 
700   /**
701    * Multiply two matrices together
702    * @function
703    * @param {Array} m1 The first operand
704    * @param {Array} m2 The second operand
705    * @returns {Array} The resulting matrix
706    * @since 0.0.0.4
707    */
708   mult : function(m1, m2) {
709     return [
710       m1[0] * m2[0] + m1[1] * m2[2],
711       m1[0] * m2[1] + m1[1] * m2[3],
712       m1[2] * m2[0] + m1[3] * m2[2],
713       m1[2] * m2[1] + m1[3] * m2[3]
714     ];
715   },
716 
717   /**
718    * Adds two matrices together
719    * @function
720    * @param {Array} m1 The first operand
721    * @param {Array} m2 The second operand
722    * @returns {Array} The resulting matrix
723    * @since 0.0.0.4
724    */
725   add : function(m1, m2) {
726     return [
727       m1[0] + m2[0],
728       m1[1] + m2[1],
729       m1[2] + m2[2],
730       m1[3] + m2[3]
731     ];
732   },
733 
734   /**
735    * Adds two matrices together and mutates the first
736    * @function
737    * @param {Array} m1 The first operand
738    * @param {Array} m2 The second operand
739    * @returns {void}
740    * @since 0.0.0.4
741    */
742   addMutate : function(m1, m2) {
743     m1[0] = m1[0] + m2[0];
744     m1[1] = m1[1] + m2[1];
745     m1[2] = m1[2] + m2[2];
746     m1[3] = m1[3] + m2[3];
747   },

749   /**
750    * Gets the entry at row i and column j
751    * @function
752    * @param {Array} m The matrix to get the entry from
753    * @param {int} i The row number, 0 being row 1
754    * @param {int} j The column number, 0 being column 1
755    * @returns {Number} The entry at row i and column j
756    * @since 0.0.0.4
757    */
758   getEntry : function(m, i, j) {
759     return m[i * 2 + j];
760   },

762   /**
763    * Determines if the two matrices are equal
764    * @function
765    * @param {Array} m1 The first operand
766    * @param {Array} m2 The second operand
767    * @returns {Boolean} True if every entry in m1 is equal to the corresponding entry in m2
768    * @since 0.0.0.4
769    */
770   equals : function(m1, m2) {
771     return m1[0] == m2[0]
772       && m1[1] == m2[1]
773       && m1[2] == m2[2]
774       && m1[3] == m2[3];
775   }
