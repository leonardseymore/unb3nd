 25 /**
 26  * @class Boundary bounce action
 27  * @constructor
 28  * @extends BoundaryAction
 29  * @since 0.0.0
 30  */
 31 function BoundaryBounce() {
 32 
 33 	/**
 34 	 * Apply this boundary action to the supplied entity
 35 	 * @function
 36 	 * @override
 37 	 * @param {Entity} entity The entity to apply this behaviour to
 38 	 * @returns {void}
 39 	 * @since 0.0.0
 40 	 */
 41 	this.update = function(entity) {
 42 		var rect = entity.boundingBox;
 43 		if (entity.pos.x < rect.x) {
 44 			entity.pos.x = rect.x;
 45 			entity.vel.x = -entity.vel.x;
 46 		} // if
 47 		
 48 		if (entity.pos.x + entity.width > rect.x + rect.width) {
 49 			entity.pos.x = rect.x + rect.width - entity.width;
 50 			entity.vel.x = -entity.vel.x;
 51 		} // if
 52 		
 53 		if (entity.pos.y < rect.y) {
 54 			entity.pos.y = rect.y;
 55 			entity.vel.y = -entity.vel.y;
 56 		} // if
 57 		
 58 		if (entity.pos.y + entity.height > rect.y + rect.height) {
 59 			entity.pos.y = rect.y + rect.height - entity.height;
 60 			entity.vel.y = -entity.vel.y;
 61 		} // if
 62 	}
 63 
 64 }
 65 BoundaryBounce.prototype = new BoundaryAction();
 66 /**
 67  * Singleton instance used to share this action between entities
 68  * @field
 69  * @static
 70  * @type BoundaryBounce
 71  * @since 0.0.0
 72  */
 73 BoundaryBounce.instance = new BoundaryBounce();

 75 /**
 76  * @class Boundary wrap action
 77  * @constructor
 78  * @extends BoundaryAction
 79  * @since 0.0.0
 80  */
 81 function BoundaryWrap() {
 82 
 83 	/**
 84 	 * Apply this boundary action to the supplied entity
 85 	 * @function
 86 	 * @override
 87 	 * @param {Entity} entity The entity to apply this behaviour to
 88 	 * @returns {void}
 89 	 * @since 0.0.0
 90 	 */
 91 	this.update = function(entity) {
 92 		var rect = entity.boundingBox;
 93 		if (entity.pos.x + entity.width < rect.x) {
 94 			entity.pos.x = rect.x + rect.width;
 95 		} // if
 96 		
 97 		if (entity.pos.x > rect.x + rect.width) {
 98 			entity.pos.x = rect.x - entity.width;
 99 		} // if
100 		
101 		if (entity.pos.y + entity.height < rect.y) {
102 			entity.pos.y = rect.y + rect.height;
103 		} // if
104 		
105 		if (entity.pos.y > rect.y + rect.height) {
106 			entity.pos.y = rect.y - entity.height;
107 		} // if
108 	}
109 
110 }

114  * @field
115  * @static

121 /**
122  * @class Boundary die action
123  * @constructor
124  * @extends BoundaryAction
125  * @since 0.0.0
126  */
127 function BoundaryDie() {
128 
129 	/**
130 	 * Apply this boundary action to the supplied entity
131 	 * @function
132 	 * @override
133 	 * @param {Entity} entity The entity to apply this behaviour to
134 	 * @returns {void}
135 	 * @since 0.0.0
136 	 */
137 	this.update = function(entity) {
138 		var rect = entity.boundingBox;
139 		if (entity.pos.x < rect.x ||
140 			entity.pos.x + entity.width > rect.x + rect.width ||
141 			entity.pos.y < rect.y ||
142 			entity.pos.y + entity.height > rect.y + rect.height) {
143 			entity.alive = false;
144 		} // if
145 	}
146 
147 }

213  /**
214  * @class Game entity having spacial and physical properties
215  * @constructor
216  * @param {Sprite} sprite The renderable presentation of this entity
217  * @since 0.0.0
218  */
219 function Entity(sprite) {
220 
221 	/**
222 	 * Flag to indicate if entity is alive
223 	 * @field 
224 	 * @type boolean
225 	 * @default true
226 	 * @since 0.0.0
227 	 */
228 	this.alive = true;
229 	
230 	/**
231 	 * Entity width
232 	 * @field 
233 	 * @type int
234 	 * @default 16
235 	 * @since 0.0.0
236 	 */
237 	this.width = 16;
238 	
239 	/**
240 	 * Entity height
241 	 * @field 
242 	 * @type int
243 	 * @default 16
244 	 * @since 0.0.0
245 	 */
246 	this.height = 16;
247 	
248 	/**
249 	 * Entity position
250 	 * @field 
251 	 * @type Vector2
252 	 * @default Zero vector
253 	 * @since 0.0.0
254 	 */
255 	this.pos = new Vector2();
256 	
257 	/**
258 	 * Entity velocity
259 	 * @field 
260 	 * @type Vector2
261 	 * @default Zero vector
262 	 * @since 0.0.0
263 	 */
264 	this.vel = new Vector2();
265 
266 	/**
267 	 * Bounding box 
268 	 * @field 
269 	 * @type Rectangle
270 	 * @default {@link windowRect}
271 	 * @since 0.0.0
272 	 */
273 	this.boundingBox = windowRect;
274 	
275 	/**
276 	 * Action to perform when boundary is breached
277 	 * @field 
278 	 * @type BoundaryAction
279 	 * @default {@link BoundaryBounce#instance}
280 	 * @since 0.0.0
281 	 */
282 	this.boundingCollisionAction = BoundaryBounce.instance;
283 
284 	/**
285 	 * Angle in radians
286 	 * @field
287 	 * @type float
288 	 * @default 0.0
289 	 * @since 0.0.0
290 	 */
291 	this.angle = 0.0;
292 	
293 	/**
294 	 * Angular velocity in radians
295 	 * @field
296 	 * @type float
297 	 * @default Math.PI
298 	 * @since 0.0.0
299 	 */
300 	this.anglev = Math.PI / 1; // angular velocity
301 	
302 	/**
303 	 * Renderable sprite
304 	 * @field
305 	 * @type Sprite
306 	 * @default undefined
307 	 * @since 0.0.0
308 	 */
309 	this.sprite = undefined;
310 	if (sprite) {
311 		this.sprite = sprite;
312 		
313 		this.width = sprite.width;
314 		this.height = sprite.height;
315 	} // if
316 	
317 	/**
318 	 * Gets the X-position
319 	 * @function
320 	 * @returns {float} The X position
321 	 * @since 0.0.0
322 	 */
323 	 this.getX = function() {
324 		return this.pos.x;
325 	 }
326 	 
327 	 /**
328 	 * Gets the Y-position
329 	 * @function
330 	 * @returns {float} The Y position
331 	 * @since 0.0.0
332 	 */
333 	 this.getY = function() {
334 		return this.pos.y;
335 	 }
336 		
337 	/**
338 	 * Update this entity
339 	 * @function
340 	 * @param {delta} Time delta in milliseconds
341 	 * @returns {void}
342 	 * @since 0.0.0
343 	 */
344 	this.update = function(delta) {
345 		var dt = delta / 1000;
346 		
347 		this.angle += this.anglev * dt;
348 		this.pos.x += this.vel.x * dt;
349 		this.pos.y += this.vel.y * dt;
350 		
351 		this.boundingCollisionAction.update(this);
352 	}
353 	
354 	/**
355 	 * Draws a placeholder for this entity
356 	 * @function
357 	 * @returns {void}
358 	 * @since 0.0.0
359 	 */
360 	this.drawPlaceholder = function() {
361 		ctx.strokeRect(0, 0, this.width, this.height);
362 
363 		ctx.beginPath();
364 		ctx.moveTo(0, 0);
365 		ctx.lineTo(this.width, this.height);
366 		ctx.moveTo(this.width, 0);
367 		ctx.lineTo(0, this.height);
368 		ctx.stroke();
369 	}
370 	
371 	/**
372 	 * Draw this entity
373 	 * @function
374 	 * @returns {void}
375 	 * @since 0.0.0
376 	 */
377 	this.draw = function() {
378 		ctx.save();	
379 		ctx.translate(this.pos.x, this.pos.y);
380 		ctx.rotate(this.angle);
381 		if (this.sprite) {
382 			this.sprite.draw();
383 		} else {
384 			this.drawPlaceholder();
385 		} // if
386 
387 		ctx.restore();
388 	}
389 }

391 /**
392  * @class Game entity manager responsible for handling multiple entities
393  * @constructor
394  * @since 0.0.0
395  */
396 function EntityManager() {
397 
398 	/**
399 	 * All managed entities
400 	 * @field
401 	 * @type Entity []
402 	 * @default []
403 	 * @since 0.0.0
404 	 */
405 	this.entities = [];
406 	
407 	/**
408 	 * Add a new entity to be managed
409 	 * @function
410 	 * @param {Entity} entity The entity to add
411 	 * @returns {int} Number of managed entities
412 	 * @since 0.0.0
413 	 */
414 	this.addEntity = function(entity) {
415 		if (debug) {
416 			console.debug("Adding new managed entity");
417 		} // if
418 		
419 		return this.entities.push(entity);
420 	}
421 	
422 	/**
423 	 * Remove a managed entity at the specified index
424 	 * @function
425 	 * @param {index} The index of the entity
426 	 * @returns {Entity} The removed entity
427 	 * @since 0.0.0
428 	 */
429 	this.removeEntityAtIndex = function(index) {
430 		if (debug) {
431 			console.debug("Removing managed entity");
432 		} // if
433 		return this.entities.splice(index, 1);
434 	}
435 	
436 	/**
437 	 * Updates all game entities
438 	 * @function
439 	 * @param {int} delta Delta time in milliseconds
440 	 * @returns {void}
441 	 * @since 0.0.0
442 	 */
443 	this.update = function(delta) {
444 		for (i in this.entities) {
445 			var entity = this.entities[i];
446 			entity.update(delta);
447 			if (!entity.alive) {
448 				this.removeEntityAtIndex(i);
449 			} // if
450 		} // for
451 	}
452 	
453 	/**
454 	 * Draw all game entities
455 	 * @function
456 	 * @returns {void}
457 	 * @since 0.0.0
458 	 */
459 	this.draw = function() {
460 		for (i in this.entities) {
461 			var entity = this.entities[i];
462 			entity.draw();
463 		} // for
464 	}
465 }
