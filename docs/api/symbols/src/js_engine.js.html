270 /**
271  * @function
272  * Reset all counters
273  * @return void
274  */
275 function engineReset() {
276 	fps = 0;
277 	lastFrame = 0;
278 	lastFPS = getTime();
279 	lastSync = getTime();
280 	engine.reset();
281 }
282 
283 /**
284  * @function
285  * Time delta
286  * @return int Time delta between time and last update time
287  */
288 function getDelta() {
289 	var time = getTime();
290 	var delta = time - lastFrame;
291 	lastFrame = time;
292 
293 	return delta;	
294 }
295 
296 /**
297  * @function
298  * Recalculate frame rate
299  * @return void
300  */
301 function updateFPS() {
302 	if (getTime() - lastFPS > 1000) {
303     avgFps = fps;
304 		fps = 0;
305 		lastFPS += 1000;
306 	} // if
307 	fps++;
308 }

323 }

439 /**
440  * @function
441  * Start the engine
442  * @return void
443  */
444 function engineStart() {
445 	engineReset();
446 	getDelta(); // clear out delta
447 	if (!running) {
448 		running = true;
449 		startGame();
450 		engineMain();
451 	} else {
452 		console.warn("Start called, but already running");
453 	} // if
454 }
455 
456 /**
457  * @function
458  * Pause the engine
459  * @return void
460  */
461 function enginePause() {
462 	paused = true;
463 	engine.pause();
464 }
465 
466 /**
467  * @function
468  * Continue the engine
469  * @return void
470  */
471 function engineContinue() {
472 	paused = false;
473 	engine.pause();
474 	engineReset();
475 	setTimeout(engineMain, 0);
476 }
477 
478 /**
479  * @function
480  * Stop the engine
481  * @return void
482  */
483 function engineStop() {
484 	if (running) {
485 		running = false;
486 		stopGame();
487 	} else {
488 		console.warn("Stop called, but not running");
489 	} // if
490 }
491 
492 /**
493  * @function
494  * Sync the framerate
495  * @return void
496  */
497 function engineSync() {
498 	var sleepTime = lastSync + (1000 / FPS) - getTime();
499 	setTimeout(engineMain, sleepTime);
500 	lastSync += (1000 / FPS);
501 }
502 
503 /**
504  * @function
505  * Main loop
506  * @return void
507  */
508 function engineMain() {
509 	if (running && !paused) {
510 		var delta = getDelta();
511 		updateGame(delta);
512 		ctx.save();
513 		renderGame();
514 		ctx.restore();
515 		updateFPS();
516 		engineSync();
517 	} // while
518 }
