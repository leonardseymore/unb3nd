395   /**
396    * @function
397    * Reset all counters
398    * @return void
399    */
400   this.engineReset = function() {
401     this.fps = 0;
402     this.lastFrame = 0;
403     this.lastFPS = unb3nd.getTime();
404     this.lastSync = unb3nd.getTime();
405     this.reset();
406   };
407 
408   /**
409    * @function
410    * Time delta
411    * @return int Time delta between time and last update time
412    */
413   this.getDelta = function() {
414     var time = unb3nd.getTime();
415     var delta = time - lastFrame;
416     this.lastFrame = time;
417 
418     return delta;
419   };
420 
421   /**
422    * @function
423    * Recalculate frame rate
424    * @return void
425    */
426   this.updateFPS = function() {
427     if (unb3nd.getTime() - this.lastFPS > 1000) {
428       this.avgFps = this.fps;
429       this.fps = 0;
430       this.lastFPS += 1000;
431     } // if
432     this.fps++;
433   };
434 
435   /**
436    * @function
437    * Start the engine
438    * @return void
439    */
440   this.engineStart = function() {
441     this.engineReset();
442     this.getDelta(); // clear out delta
443     if (!this.running) {
444       this.running = true;
445       this.startGame();
446       this.engineMain();
447     } else {
448       console.warn("Start called, but already running");
449     } // if
450   };
451 
452   /**
453    * @function
454    * Pause the engine
455    * @return void
456    */
457   this.enginePause = function() {
458     this.paused = true;
459     engine.pause();
460   };
461 
462   /**
463    * @function
464    * Continue the engine
465    * @return void
466    */
467   this.engineContinue = function() {
468     this.paused = false;
469     engine.pause();
470     this.engineReset();
471     setTimeout(String | Function(this.engineMain), 0);
472   };
473 
474   /**
475    * @function
476    * Stop the engine
477    * @return void
478    */
479   this.engineStop = function() {
480     if (this.running) {
481       this.running = false;
482       this.stopGame();
483     } else {
484       console.warn("Stop called, but not running");
485     } // if
486   };
487 
488   /**
489    * @function
490    * Sync the framerate
491    * @return void
492    */
493   this.engineSync = function() {
494     var sleepTime = this.lastSync + (1000 / FPS) - unb3nd.getTime();
495     setTimeout(String | Function(this.engineMain), sleepTime);
496     this.lastSync += (1000 / FPS);
497   };
498 
499   /**
500    * @function
501    * Main loop
502    * @return void
503    */
504   this.engineMain = function() {
505     if (this.running && !this.paused) {
506       var delta = this.getDelta();
507       this.updateGame(delta);
508       ctx.save();
509       this.renderGame();
510       ctx.restore();
511       this.updateFPS();
512       this.engineSync();
513     } // while
514   };
515 }

519 /*

522  * @link{#init} method
523  */
524 document.onreadystatechange = function () {
525   if (document.readyState == "complete") {
526     if (engine.debug) {
527       console.debug("Document ready state changed to 'complete'");
528     } // if
529 
530     engine.engineInit();
531   } // if
532 }

539 document.onkeydown = function (e) {
540   // toggle debug 'D'
541   if (e.keyCode == 68) {
542     engine.debug = !engine.debug;
543     console.debug("Debug enabled: " + engine.debug);
544   } // if
545 
546   engine.keydown(e);
547 }
