  7 /**
  8  * @class Background base class
  9  * @constructor
 10  * @since 0.0.0
 11  */
 12 function Background() {
 13 
 14 	/**
 15 	 * Update the background
 16 	 * @function
 17 	 * @param {int} delta Delta time in milliseconds
 18 	 * @returns {void}
 19 	 * @since 0.0.0
 20 	 */
 21 	this.update = function(delta) {}
 22 	
 23 	/**
 24 	 * Draw the background
 25 	 * @function
 26 	 * @param {Rectangle} area The area to draw the background on
 27 	 * @returns {void}
 28 	 * @since 0.0.0
 29 	 */
 30 	this.draw = function(area) {}
 31 }
 32 
 33 /**
 34  * @class A simple background that takes a fill style for the background
 35  * @constructor
 36  * @extends Background
 37  * @param {FillStyle} fillStyle The style to use to fill the background
 38  * @since 0.0.0
 39  */
 40 function FillBackground(fillStyle) {
 41 
 42 	/**
 43 	 * Fill style to flood the background
 44 	 * @field 
 45 	 * @type FillStyle
 46 	 * @default fillStyle
 47 	 * @since 0.0.0
 48 	 */
 49 	this.fillStyle = fillStyle;
 50 	
 51 	/**
 52 	 * Draw the background
 53 	 * @function
 54 	 * @override
 55 	 * @param {Rectangle} area The area to draw the background on
 56 	 * @returns {void}
 57 	 * @since 0.0.0
 58 	 */
 59 	this.draw = function(area) {
 60 		ctx.fillStyle = this.fillStyle;
 61 		ctx.fillRect(area.x, area.y, area.width, area.height);
 62 	}
 63 }
 64 FillBackground.prototype = new Background();

 66 /**
 67  * @class An animated starry sky background
 68  * @constructor
 69  * @extends Background
 70  * @param {int} numStars Number of stars
 71  * @param {FillStyle} fillStyle Basic fill style
 72  * @since 0.0.0
 73  */
 74 function StarryBackground(numStars, fillStyle) {
 75 
 76 	/**
 77 	 * @class A star
 78 	 * @constructor
 79 	 * @param {float} x X-coordinate
 80 	 * @param {float} y Y-coordinate
 81 	 * @param {float} i Intensity 0 to 1
 82 	 * @since 0.0.0
 83 	 */
 84 	function Star(x, y, i) {
 85 	
 86 		/**
 87 		 * X-coordinate
 88 		 * @field 
 89 		 * @type float
 90 		 * @default x
 91 		 * @since 0.0.0
 92 		 */
 93 		this.x = x;
 94 		
 95 		/**
 96 		 * Y-coordinate
 97 		 * @field 
 98 		 * @type float
 99 		 * @default y
100 		 * @since 0.0.0
101 		 */
102 		this.y = y;
103 		
104 		/**
105 		 * Intensity 0.0 to 1.0
106 		 * @field 
107 		 * @type float
108 		 * @default i
109 		 * @since 0.0.0
110 		 */
111 		this.i = i;
112 		
113 		/**
114 		 * Star radius
115 		 * @field 
116 		 * @type float
117 		 * @default Random[2,6]
118 		 * @since 0.0.0
119 		 */
120 		this.r = Math.floor(Math.random() * 4) + 2;
121 		
122 		/**
123 		 * Animate this star over the give time delta
124 		 * @function
125 		 * @param {int} delta Delta time in milliseconds
126 		 * @returns {void}
127 		 * @since 0.0.0
128 		 */
129 		this.animate = function(delta) {
130 			if (Math.random() < 0.005) {
131 				this.i = Math.random();
132 			} // if
133 		}
134 		
135 		/**
136 		 * Draw the star as a pixel
137 		 * @function
138 		 * @returns {void}
139 		 * @since 0.0.0
140 		 */
141 		this.draw = function() {
142 			ctx.fillStyle = "rgba(255, 255, 255, " + this.i + ")";
143 			ctx.fillRect(this.x, this.y, 1, 1);
144 		}
145 		
146 		/**
147 		 * Draw the star as a cartoon styled star
148 		 * @function
149 		 * @returns {void}
150 		 * @since 0.0.0
151 		 */
152 		this.drawCartoon = function() {
153 			ctx.save();
154 			ctx.fillStyle = "rgba(255, 255, 255, " + this.i + ")";
155 			ctx.translate(this.x, this.y);
156 			ctx.beginPath()
157 			ctx.moveTo(this.r, 0);
158 			for (var i = 0; i < 9; i++){
159 				ctx.rotate(Math.PI / 5);
160 				if(i % 2 == 0) {
161 					ctx.lineTo((this.r / 0.525731) * 0.200811,0);
162 				} else {
163 					ctx.lineTo(this.r, 0);
164 				} // if
165 			} // for
166 			ctx.closePath();
167 			ctx.fill();
168 			ctx.restore();
169 		}
170 	}
171 
172 	/**
173 	 * Cartoonify stars
174 	 * @field 
175 	 * @type boolean
176 	 * @default true
177 	 * @since 0.0.0
178 	 */ 
179 	this.cartoonify = true;
180 	
181 	/**
182 	 * Fill style to flood the background
183 	 * @field 
184 	 * @type FillStyle
185 	 * @default {@link STYLE_DARK_EVENING}
186 	 * @since 0.0.0
187 	 */ 
188 	this.fillStyle = fillStyle || STYLE_DARK_EVENING;
189 	
190 	/**
191 	 * Number of stars
192 	 * @field 
193 	 * @type int
194 	 * @default 50
195 	 * @since 0.0.0
196 	 */ 
197 	this.numStars = numStars || 50;
198 	
199 	/**
200 	 * Very lucky if you see one of these
201 	 * @field 
202 	 * @type Star
203 	 * @default undefined
204 	 * @since 0.0.0
205 	 */ 
206 	this.shootingStar = undefined;
207 	
208 	/**
209 	 * Star intensities
210 	 * @field 
211 	 * @type float []
212 	 * @default []
213 	 * @since 0.0.0
214 	 */ 
215 	this.stars = [];
216 	for (i = 0; i < numStars; i++) {
217 		var star = new Star(windowRect.width * Math.random(), 
218 			windowRect.height * Math.random(), 
219 			Math.random());
220 		this.stars.push(star);
221 	}
222 	
223 	/**
224 	 * Update the background
225 	 * @function
226 	 * @override
227 	 * @param {int} delta Delta time in milliseconds
228 	 * @returns {void}
229 	 * @since 0.0.0
230 	 */
231 	this.update = function(delta) {
232 		var dt = delta / 1000;
233 		if (!this.shootingStar) {
234 			if (Math.random() < 0.002) {
235 				this.shootingStar = new Star(0, Math.random() * windowRect.height, Math.random());
236 			} // if
237 		} else {
238 			this.shootingStar.x += 500 * dt;
239 			this.shootingStar.y += 300 * dt;
240 			if (this.shootingStar.x > windowRect.width) {
241 				this.shootingStar = undefined;
242 			} // if
243 		} // if
244 		
245 		for (i = 0; i < numStars; i++) {
246 			var star = this.stars[i];
247 			star.animate(delta);
248 		} // for
249 	}
250 	
251 	/**
252 	 * Draw the background
253 	 * @function
254 	 * @override
255 	 * @param {Rectangle} area The area to draw the background on
256 	 * @returns {void}
257 	 * @since 0.0.0
258 	 */
259 	this.draw = function(area) {
260 		ctx.save();
261 		ctx.fillStyle = this.fillStyle;
262 		ctx.fillRect(area.x, area.y, area.width, area.height);
263 		for (i = 0; i < numStars; i++) {
264 			var star = this.stars[i];
265 			if (this.cartoonify) {
266 				star.drawCartoon();
267 			} else {
268 				star.draw();
269 			} // if
270 		} // for
271 		
272 		if (this.shootingStar) {
273 			if (this.cartoonify) {
274 				this.shootingStar.drawCartoon();
275 			} else {
276 				this.shootingStar.draw();
277 			} // if
278 		} // if		
279 		ctx.restore();
280 	}
281 }
282 StarryBackground.prototype = new Background();
