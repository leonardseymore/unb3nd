  7 /** 
  8  * @class Resources loader
  9  * @constructor
 10  * @since 0.0.0
 11  */
 12 function Resources() {
 13 }
 14 
 15 /** 
 16  * Resources configuration must be configured per application to be able 
 17  * to locate resources.
 18  * @field
 19  * @static
 20  * @type Object
 21  * @default Fully initialized static
 22  * @since 0.0.0
 23  */
 24 Resources.configuration = {
 25 	ROOT_PATH : "../",
 26 	IMG_PATH : "../img/"
 27 };
 28 
 29 /**
 30  * Loads an image resource from name
 31  * TODO: An idea is to later combine all images into a sprite sheet and this method
 32  *       could possibly retrieve it from the spritesheet rather than file.
 33  * @function
 34  * @static
 35  * @param {string} name The name of the image to load
 36  * @return {Image} The loaded image
 37  * @since 0.0.0
 38  * @see Resources#configuration
 39  */
 40 Resources.loadImage = function(name) {
 41 	var image = new Image();
 42 	image.src = Resources.configuration.IMG_PATH + name;
 43 	return image;
 44 }
